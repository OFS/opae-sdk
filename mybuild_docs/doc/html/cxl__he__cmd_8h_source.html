60  void he_perf_counters(he_cxl_latency cxl_latency = HE_CXL_LATENCY_NONE) {
61  volatile he_cache_dsm_status *dsm_status = NULL;
62 
63  dsm_status = reinterpret_cast<he_cache_dsm_status *>(
64  (uint8_t *)(host_exe_->get_dsm()));
65  if (!dsm_status)
66  return;
67 
68  cout << "\n********* DSM Status CSR Start *********" << endl;
69  cout << "test completed :" << dsm_status->test_completed << endl;
70  cout << "dsm number:" << dsm_status->dsm_number << endl;
71  cout << "error vector:" << dsm_status->err_vector << endl;
72  cout << "num ticks:" << dsm_status->num_ticks << endl;
73  cout << "num reads:" << dsm_status->num_reads << endl;
74  cout << "num writes:" << dsm_status->num_writes << endl;
75  cout << "penalty start:" << dsm_status->penalty_start << endl;
76  cout << "penalty end:" << dsm_status->penalty_end << endl;
77  cout << "actual data:" << dsm_status->actual_data << endl;
78  cout << "expected data:" << dsm_status->expected_data << endl;
79 
80  double latency = 0;
81  double perf_data = 0;
82  // print bandwidth
83  if (dsm_status->num_ticks > 0) {
84  perf_data = he_num_xfers_to_bw(dsm_status->num_reads +
85  dsm_status->num_writes, dsm_status->num_ticks);
86 
87  if (cxl_latency == HE_CXL_RD_LATENCY) {
88  //To convert clock ticks to nanoseconds,multiply the clock ticks by 2.5
89  latency = (double)(dsm_status->num_ticks * 2.5);
90  host_exe_->logger_->info("Bandwidth: {0:0.3f} GB/s Total transaction time: {1:0.2f} nanoseconds",
91  perf_data, latency);
92  } else {
93  host_exe_->logger_->info("Bandwidth: {0:0.3f} GB/s", perf_data);
94  }
95  } else {
96  host_exe_->logger_->info("Read Latency: N/A");
97  }
98 
99  cout << "********* DSM Status CSR end *********" << endl;
100  }

243  bool he_set_bias_mode() {
244 
245  // Target memory HOST set BIAS host
246  if (he_target_ == HE_TARGET_HOST) {
247  he_ctl_.bias_support = HOSTMEM_BIAS;
248  // Target memory FPGA set BIAS host/device
249  if (he_bias_ == HOSTMEM_BIAS) {
250  he_ctl_.bias_support = HOSTMEM_BIAS;
251  } else {
252  cerr << "Wrong BIAS mode for specified target memory type" << endl;
253  return false;
254  }
255  } else {
256  // Target memory FPGA set BIAS host/device
257  if (he_bias_ == HOSTMEM_BIAS) {
258  he_ctl_.bias_support = FPGAMEM_HOST_BIAS;
259  } else {
260  he_ctl_.bias_support = FPGAMEM_DEVICE_BIAS;
261  }
262  }
263 
264  return true;
