44 #define CXL_TG_BW_FACTOR 0.931323

46 /*
47 1) Write to TG_CLEAR with data=0xF to clear all the failure status registers.
48 2) Configure the registers with the value specified in table 1 below.
49 3) Write to TG_START to start the TG2 using the configuration in step 2.
50 This starts the traffic test in user mode.
51 4) Read from TG_TEST_COMPLETE until the read data =0x1,
52 indicating the traffic test has completed.
53 5) Read from TG_PASS, TG_FAIL, and TG_TIMEOUT to check the test result.
54  a) TG_PASS. A value of 1 indicates that the traffic
55  test passed at the end of all test stages.
56  b) TG_FAIL. A value of 1 indicates that the configured
57  traffic finished running but a failure (read miscompare) was observed. You
58 may read from other relevant registers to get more information about the
59 failure. Refer to the Configuration and Status Registers table for information
60 on the available registers. c) TG_TIMEOUT. A value of 1 indicates that a read
61  response was not received from the interface for one or more read commands.
62 */

207  bool tg_wait_test_completion() {
208  /* Wait for test completion */
209  uint32_t timeout = MEM_TG_TEST_TIMEOUT;
210  uint64_t tg_status = 0x0;
211  uint32_t tg_fail = 0;
212  tg_exe_->logger_->debug("tg wait for test completion...");
213 
214  tg_status = tg_exe_->read32(TG_TEST_COMPLETE);
215  tg_exe_->logger_->debug("test complete status:{}", tg_status);
216 
217  while (tg_status != 0x1) {
218  tg_status = tg_exe_->read32(TG_TEST_COMPLETE);
219  std::this_thread::yield();
220  if (--timeout == 0) {
221  std::cerr << "test completion timeout " << std::endl;
222  tg_exe_->status_ = -1;
223  tg_print_fail_info();
224  return false;
225  }
226  }
227 
228  tg_exe_->logger_->debug("tg pass:{}", tg_exe_->read32(TG_PASS));
229 
230  tg_fail = tg_exe_->read32(TG_FAIL);
231  if (tg_fail == 0x1) {
232  std::cerr << "Tg test failed:" << tg_fail << std::endl;
233  tg_exe_->status_ = -1;
234  tg_print_fail_info();
235  return false;
236  }
237 
238  // poll while active bit is set (channel status =
239  // {pass,fail,timeout,active})
240  tg_status = 0x0;
241  tg_status = 0xF & (tg_exe_->read64(MEM_TG_STAT));
242  tg_exe_->logger_->debug("Memory Traffic Generator Status:{}",
243  tg_exe_->read64(MEM_TG_STAT));
244 
245  while (tg_status == TG_STATUS_ACTIVE) {
246  tg_status = 0xF & (tg_exe_->read64(MEM_TG_STAT));
247  std::this_thread::yield();
248  if (--timeout == 0) {
249  std::cerr << "Memory Traffic Generator status active" << std::endl;
250  tg_exe_->status_ = -1;
251  return false;
252  }
253  }
254  if (tg_status == TG_STATUS_TIMEOUT) {
255  std::cerr << "Memory Traffic Generator status timeout" << std::endl;
256  tg_exe_->status_ = -1;
257  return false;
258  }
259 
260  if (tg_status == TG_STATUS_ERROR) {
261  std::cerr << "Memory Traffic Generator status error" << std::endl;
262  tg_exe_->status_ = -1;
263  return false;
264  }
265  tg_exe_->status_ = tg_status;
266 
267  tg_exe_->logger_->debug("Memory Traffic Generator Status pass");
268  return true;
269  }
