228  virtual int run(CLI::App *app, test_command::ptr_t test) override {
229  int res = exit_codes::not_run;
230  logger_->info("starting test run, count of {0:d}", count_);
231  uint32_t count = 0;
232  try {
233  while (count < count_) {
234  logger_->debug("starting iteration: {0:d}", count + 1);
235  res = test_afu::run(app, test);
236  count++;
237  logger_->debug("end iteration: {0:d}", count);
238  if (res) break;
239  }
240  } catch (std::exception &ex) {
241  logger_->error(ex.what());
242  res = exit_codes::exception;
243  }
244  auto pass = res == exit_codes::success ? "PASS" : "FAIL";
245  logger_->info("Test {}({}): {}", test->name(), count, pass);
246  spdlog::drop_all();
247  return res;
248  }
