31 // The intention of this utility is to demonstrate simple use of the 200G/400G
32 // Ethernet subsystem so that a user may add their own complexity on top.
33 
34 // Code structure:
35 // This class inherits hssi_cmd (which in turn inherits 'command') as part of
36 // the afu-test framework. It implements a specific 'command' (which is
37 // essentially like a test / sequence of operations) to exercise the 200G-400G
38 // F-Tile HSSI Sub-System.
39 
40 // HSSI = "High Speed Serial Interface" which is better referred to as
41 // "Ethernet". It expects a specific AFU (accelerator function unit) to be
42 // present in the FPGA (specified by afu_id()). This corresponding AFU
43 // instantiates a traffic generator (TG) to transmit/receive packet data to the
44 // HSSI-SS.
45 
46 // The run() function talks to the TG to create some test traffic flow and
47 // measure throughput.
48 
49 // The TG has very limited control -- you can only control the number of packets
50 // sent. The following terms are synonymous: traffic generator, traffic
51 // controller, packet client. The TG has its own control/status register (CSR)
52 // map that is abstracted behind a "Mailbox". The Mailbox presents a small set
53 // of command registers that accept address, data, control. Through these, the
54 // user can read/write the abstracted register space. The Mailbox CSR address
55 // space is defined with the hssi_afu() class. The Mailbox may have multiple
56 // ports (also called 'channels') where each port provides a window into a
57 // distinct abstracted address space. In other words, the AFU may instantiate
58 // multiple TGs where each TG is accessed through a separate Mailbox port.
59 
60 // It is assumed that a 200G AFU uses two Mailbox channels (i.e. two traffic
61 // generators) and a 400G AFU uses one channel.
62 
63 // Registers in the AFU CSR space. Other registers, in this space, are defined
64 // in hssi_afu.h
65 #define CSR_AFU_400G_TG_EN 0x0058
66 
67 // 200G/400G traffic generator registers. This address space is accessed
68 // indirectly through the Mailbox The CSR map is defined in a spreadsheet,
69 // located here at the time of this writing:
70 // https://github.com/OFS/ofs-fim-common/tree/release/ofs-2023.2/src/common/he_hssi/HE_HSSI_TC_200_400G.xls
71

111  virtual int run(test_afu *afu, CLI::App *app) override {
112  (void)app;
113 
114  hssi_afu *hafu = dynamic_cast<hssi_afu *>(afu);
115 
116  // Check if this AFU is for 200G or 400G by reading a status reg in the AFU
117  // CSR region.
118  bool tg_200n_400;
119  tg_200n_400 = bool(
120  hafu->read64(CSR_AFU_400G_TG_EN)); // Bit-0 = 0 for 200g, 1 for 400g
121  std::cout << "Detected " << (tg_200n_400 ? "400G" : "200G") << std::setw(21)
122  << " HE-HSSI AFU." << std::endl
123  << std::endl;
124 
125  // For 200G there are two Ethernet ports, #8 and #12.
126  // For 400G there is one Ethernet port, #8.
127  // These are connected to Mailbox channels 0 and 1, respectively.
128  int num_ports = tg_200n_400 ? 1 : 2;
129  for (int i = 0; i < num_ports; i++) {
130  // Select the appropriate port on the Mailbox
131  std::cout << "Setting traffic control/mailbox channel-select to " << i
132  << std::endl;
133  hafu->write64(TRAFFIC_CTRL_PORT_SEL, i);
134 
135  // Set ROM start/end address.
136  // The TG reads and transmits packet data from a 1024-word ROM. The ROM
137  // contents are fixed at FPGA compile-time. It contains packets of size
138  // 1482-bytes. Each loop through the ROM transmits 32 packets. The
139  // end-address (i.e. the final address to read from in each loop through
140  // the ROM) is different in 200G vs 400G. CSR_HW_TEST_ROM_ADDR[15:0] =
141  // ROM start address CSR_HW_TEST_ROM_ADDR[31:16] = ROM end address
142  std::cout << "Setting start/end ROM address" << std::endl;
143  uint32_t reg;
144  reg = 0;
145  reg |= tg_200n_400 ? (0x0191 << 16) : (0x02FF << 16);
146  hafu->mbox_write(CSR_HW_TEST_ROM_ADDR, reg);
147 
148  uint32_t num_packets_per_rom_loop = 32;
149  if ((num_packets_ % num_packets_per_rom_loop != 0) ||
150  (num_packets_ < num_packets_per_rom_loop)) {
151  std::cout << "--num_packets <num> must be >="
152  << num_packets_per_rom_loop << std::setw(21)
153  << " and a multiple of " << num_packets_per_rom_loop
154  << std::setw(21)
155  << " since the traffic generator only sends this multiple."
156  << std::endl;
157  return test_afu::error;
158  }
159 
160  uint32_t rom_loop_count = num_packets_ / num_packets_per_rom_loop;
161  // The ROM loopcount register in the TG is 32-bits. Since num_packets_ is
162  // itself only 32 bits, it's guaranteed we will not overflow
163  // rom_loop_count.
164 
165  reg = rom_loop_count;
166  std::cout << "num_packets = " << num_packets_
167  << ". Setting ROM loop count to " << reg << std::endl;
168  std::cout << "Packet length is fixed at 1482-bytes. 32 packets are sent "
169  "for every ROM loop."
170  << std::endl;
171  hafu->mbox_write(CSR_HW_TEST_LOOP_CNT, reg);
172 
173  std::cout << "Resetting traffic-generator counters" << std::endl;
174  reg = 0x180; // Set bit-7 = 1 (clear status regs), bit8 = 1 (clear
175  // counters themselves).
176  hafu->mbox_write(CSR_HW_PC_CTRL, reg);
177  reg = 0x0; // bit-7 must be manually cleared. bit-8 is self-clearing.
178  hafu->mbox_write(CSR_HW_PC_CTRL, reg);
179 
180  std::cout << "Starting the traffic-generator." << std::endl;
181  // Start the TG
182  reg = 0x1;
183  hafu->mbox_write(CSR_HW_PC_CTRL, reg);
184 
185  std::cout << "Waiting for all packets to be sent." << std::endl;
186  // Loop until the TX SOP Count matches the expected num_packets.
187  uint64_t tx_sop_count = 0;
188  const uint64_t interval = 100ULL;
189  while (tx_sop_count < num_packets_) {
190  tx_sop_count =
191  ((uint64_t)hafu->mbox_read(CSR_STAT_TX_SOP_CNT_MSB) << 32) |
192  (uint64_t)hafu->mbox_read(CSR_STAT_TX_SOP_CNT_LSB);
193  if (!running()) {
194  reg = 0x00; // Stop the TG
195  hafu->mbox_write(CSR_HW_PC_CTRL, reg);
196  return test_afu::error;
197  }
198  std::this_thread::sleep_for(std::chrono::microseconds(interval));
199  }
200 
201  std::cout << "Stopping the traffic-generator" << std::endl;
202  reg = 0x00; // Stop the TG (bit-0=0) and take snapshot (bit-6=1)
203  hafu->mbox_write(CSR_HW_PC_CTRL, reg);
204 
205  std::cout << "Short sleep to allow packets to propagate" << std::endl;
206  sleep(1);
207  std::cout << "Taking snapshot of counters" << std::endl;
208 
209  reg = 0x40; // Take snapshot (bit-6=1)
210  hafu->mbox_write(CSR_HW_PC_CTRL, reg);
211 
212  std::cout << std::endl << std::endl;
213  print_registers(std::cout, hafu);
214  std::cout << std::endl << std::endl;
215 
216  // Throughput calculation
217  double sample_period_ns = 1000 / USER_CLKFREQ_N6001;
218  uint64_t timestamp_start, timestamp_end, timestamp_duration_cycles;
219  double timestamp_duration_ns;
220  timestamp_start =
221  ((uint64_t)hafu->mbox_read(CSR_STAT_TIMESTAMP_TG_START_MSB) << 32) |
222  (uint64_t)hafu->mbox_read(CSR_STAT_TIMESTAMP_TG_START_LSB);
223  timestamp_end =
224  ((uint64_t)hafu->mbox_read(CSR_STAT_TIMESTAMP_TG_END_MSB) << 32) |
225  (uint64_t)hafu->mbox_read(CSR_STAT_TIMESTAMP_TG_END_LSB);
226  assert(timestamp_end > timestamp_start);
227  timestamp_duration_cycles = timestamp_end - timestamp_start;
228  timestamp_duration_ns = timestamp_duration_cycles * sample_period_ns;
229  uint64_t payload_bytes_per_packet = 1464;
230  uint64_t total_bytes_per_packet = 1482;
231  uint64_t total_num_packets = num_packets_per_rom_loop * rom_loop_count;
232  uint64_t total_payload_size_bytes =
233  payload_bytes_per_packet * total_num_packets;
234  uint64_t total_size_bytes = total_bytes_per_packet * total_num_packets;
235  double throughput_payload_bytes_gbps =
236  total_payload_size_bytes / timestamp_duration_ns;
237  double throughput_total_bytes_gbps =
238  total_size_bytes / timestamp_duration_ns;
239 
240  std::cout << "AFU clock frequency : " << USER_CLKFREQ_N6001 << " MHz"
241  << std::endl
242  << "Total # packets transmitted: " << total_num_packets
243  << std::endl
244  << "Total payload transmitted: " << total_payload_size_bytes
245  << " bytes" << std::endl
246  << "Total data transmitted: " << total_size_bytes << " bytes"
247  << std::endl
248  << "Total duration ns: " << timestamp_duration_ns << " ns."
249  << std::endl
250  << "Throughput on payload data: "
251  << throughput_payload_bytes_gbps
252  << " Gbytes/sec = " << throughput_payload_bytes_gbps * 8
253  << " Gbits/sec" << std::endl
254  << "Throughput on total data: " << throughput_total_bytes_gbps
255  << " Gbytes/sec = " << throughput_total_bytes_gbps * 8
256  << " Gbits/sec" << std::endl
257  << std::endl
258  << std::endl
259  << std::endl;
260  }
261 
262  return test_afu::success;
263  }
