152  virtual int run(test_afu *afu, CLI::App *app) override
153  {
154  (void)app;
155 
156  hssi_afu *hafu = dynamic_cast<hssi_afu *>(afu);
157 
158  uint64_t bin_src_addr = mac_bits_for(src_addr_);
159  if (bin_src_addr == INVALID_MAC) {
160  std::cerr << "invalid MAC address: " << src_addr_ << std::endl;
161  return test_afu::error;
162  }
163 
164  uint64_t bin_dest_addr = mac_bits_for(dest_addr_);
165  if (bin_dest_addr == INVALID_MAC) {
166  std::cerr << "invalid MAC address: " << dest_addr_ << std::endl;
167  return test_afu::error;
168  }
169 
170  std::string eth_ifc = eth_ifc_;
171  if (eth_ifc_ == "none")
172  eth_ifc = hafu->ethernet_interface();
173 
174  if (port_ != INVALID_PORT) {
175  std::cout << "--port is overriding --src-port and --dst-port" << std::endl;
176  src_port_ = dst_port_ = port_;
177  }
178 
179  std::cout << "10G loopback test" << std::endl
180  << " Tx/Rx port: " << port_ << std::endl
181  << " Tx port: " << src_port_ << std::endl
182  << " Rx port: " << dst_port_ << std::endl
183  << " eth_loopback: " << eth_loopback_ << std::endl
184  << " he_loopback: " << he_loopback_ << std::endl
185  << " num_packets: " << num_packets_ << std::endl
186  << " packet_length: " << packet_length_ << std::endl
187  << " src_address: " << src_addr_ << std::endl
188  << " (bits): 0x" << std::hex << bin_src_addr << std::endl
189  << " dest_address: " << dest_addr_ << std::endl
190  << " (bits): 0x" << std::hex << bin_dest_addr << std::endl
191  << " random_length: " << random_length_ << std::endl
192  << " random_payload: " << random_payload_ << std::endl
193  << " rnd_seed0: " << rnd_seed0_ << std::endl
194  << " rnd_seed1: " << rnd_seed1_ << std::endl
195  << " rnd_seed2: " << rnd_seed2_ << std::endl
196  << " eth: " << eth_ifc << std::endl
197  << std::endl;
198 
199  if (eth_ifc == "") {
200  std::cout << "No eth interface, so not "
201  "honoring --eth-loopback." << std::endl;
202  } else {
203  if (eth_loopback_ == "on")
204  enable_eth_loopback(eth_ifc, true);
205  else
206  enable_eth_loopback(eth_ifc, false);
207  }
208 
209  hafu->mbox_write(src_port_, CSR_STOP, 0);
210 
211  if (he_loopback_ != "none") {
212  hafu->mbox_write(src_port_, CSR_MAC_LOOP, (he_loopback_ == "on") ? 1 : 0);
213 
214  if (he_loopback_ != "on") // don't loop for "off"
215  return test_afu::success;
216 
217  std::cout << "HE loopback enabled. Use Ctrl+C to exit." << std::endl;
218 
219  while (running()) {
220  std::this_thread::sleep_for(std::chrono::milliseconds(1000));
221  }
222 
223  return test_afu::success;
224  }
225 
226  double clk_freq = clock_freq_for(hafu);
227 
228  hafu->mbox_write(src_port_, CSR_NUM_PACKETS, num_packets_);
229 
230  hafu->mbox_write(src_port_, CSR_PACKET_LENGTH, packet_length_);
231 
232  hafu->mbox_write(src_port_, CSR_SRC_ADDR0, static_cast<uint32_t>(bin_src_addr));
233  hafu->mbox_write(src_port_, CSR_SRC_ADDR1, static_cast<uint32_t>(bin_src_addr >> 32));
234 
235  hafu->mbox_write(src_port_, CSR_DEST_ADDR0, static_cast<uint32_t>(bin_dest_addr));
236  hafu->mbox_write(src_port_, CSR_DEST_ADDR1, static_cast<uint32_t>(bin_dest_addr >> 32));
237 
238  hafu->mbox_write(src_port_, CSR_RANDOM_LENGTH, (random_length_ == "fixed") ? 0 : 1);
239  hafu->mbox_write(src_port_, CSR_RANDOM_PAYLOAD, (random_payload_ == "incremental") ? 0 : 1);
240  hafu->mbox_write(src_port_, CSR_RND_SEED0, rnd_seed0_);
241  hafu->mbox_write(src_port_, CSR_RND_SEED1, rnd_seed1_);
242  hafu->mbox_write(src_port_, CSR_RND_SEED2, rnd_seed2_);
243 
244  hafu->mbox_write(src_port_, CSR_START, 1);
245 
246  print_registers(std::cout, hafu);
247 
248  uint32_t count;
249  const uint64_t interval = 100ULL;
250  do
251  {
252  count = hafu->mbox_read(src_port_, CSR_PACKET_TX_COUNT);
253 
254  if (!running()) {
255  hafu->mbox_write(src_port_, CSR_STOP, 1);
256  return test_afu::error;
257  }
258 
259  std::this_thread::sleep_for(std::chrono::microseconds(interval));
260  } while(count < num_packets_);
261 
262  std::cout << std::endl;
263 
264  if (clk_freq == INVALID_CLOCK_FREQ) {
265  std::cerr << "Couldn't determine user clock freq." << std::endl
266  << "Skipping performance display." << std::endl;
267  } else {
268  std::cout << "HSSI performance: " << std::endl;
269  // Read traffic control Tx/Rx timestamp registers
270  uint32_t tx_sta_tstamp = hafu->mbox_read(src_port_, CSR_TX_STA_TSTAMP);
271  uint32_t tx_end_tstamp = hafu->mbox_read(src_port_, CSR_TX_END_TSTAMP);
272  uint32_t rx_sta_tstamp = hafu->mbox_read(dst_port_, CSR_RX_STA_TSTAMP);
273  uint32_t rx_end_tstamp = hafu->mbox_read(dst_port_, CSR_RX_END_TSTAMP);
274 
275  // Convert timestamp register from clock cycles to nanoseconds
276  double sample_period_ns = 1000 / clk_freq;
277  double tx_sta_tstamp_ns = tx_sta_tstamp * sample_period_ns;
278  double tx_end_tstamp_ns = tx_end_tstamp * sample_period_ns;
279  double rx_sta_tstamp_ns = rx_sta_tstamp * sample_period_ns;
280  double rx_end_tstamp_ns = rx_end_tstamp * sample_period_ns;
281 
282  // Calculate latencies
283  double latency_min_ns = rx_sta_tstamp_ns - tx_sta_tstamp_ns;
284  double latency_max_ns = rx_end_tstamp_ns - tx_end_tstamp_ns;
285 
286  // Calculate Tx/Rx throughput achieved
287  uint32_t data_pkt_size;
288  if (clk_freq == USER_CLKFREQ_S10)
289  data_pkt_size = packet_length_ - 4;
290  else
291  data_pkt_size = packet_length_ - 8;
292  double total_tx_duration_ns = (tx_end_tstamp_ns - tx_sta_tstamp_ns);
293  double total_rx_duration_ns = (rx_end_tstamp_ns - rx_sta_tstamp_ns);
294  uint32_t total_data_size_bits = (num_packets_ * data_pkt_size * 8) ;
295  double achieved_tx_tput_gbps = (total_data_size_bits / total_tx_duration_ns);
296  double achieved_rx_tput_gbps = (total_data_size_bits / total_rx_duration_ns);
297 
298  std::cout << "\tSelected clock frequency : "<< clk_freq << " MHz" << std::endl
299  << "\tLatency minimum : " << latency_min_ns << " ns" <<std::endl
300  << "\tLatency maximum : " << latency_max_ns << " ns" <<std::endl
301  << "\tAchieved Tx throughput : " << achieved_tx_tput_gbps << " GB/s" << std::endl;
302  if (eth_loopback_ == "on") {
303  std::cout << "\tAchieved Rx throughput : " << achieved_rx_tput_gbps << " GB/s" << std::endl;
304  }
305  std::cout << std::endl;
306  }
307 
308  if (eth_ifc == "") {
309  std::cout << "No eth interface, so not "
310  "showing stats." << std::endl;
311  } else {
312  show_eth_stats(eth_ifc);
313 
314  if (eth_loopback_ == "on")
315  enable_eth_loopback(eth_ifc, false);
316  }
317 
318  return test_afu::success;
319  }
