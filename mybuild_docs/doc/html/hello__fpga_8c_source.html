155 #define GETOPT_STRING "hscv"
156 fpga_result parse_args(int argc, char *argv[])
157 {
158  struct option longopts[] = {
159  { "help", no_argument, NULL, 'h' },
160  { "shared", no_argument, NULL, 's' },
161  { "n3000", no_argument, NULL, 'c' },
162  { "version", no_argument, NULL, 'v' },
163  { NULL, 0, NULL, 0 }
164  };
165 
166  int getopt_ret;
167  int option_index;
168 
169  char version[32];
170  char build[32];
171 
172  while (-1 != (getopt_ret = getopt_long(argc, argv, GETOPT_STRING,
173  longopts, &option_index))) {
174  const char *tmp_optarg = optarg;
175  /* Checks to see if optarg is null and if not it goes to value of optarg */
176  if ((optarg) && ('=' == *tmp_optarg)) {
177  ++tmp_optarg;
178  }
179 
180  switch (getopt_ret) {
181  case 'h':
182  help();
183  return -1;
184  case 's':
185  config.open_flags |= FPGA_OPEN_SHARED;
186  break;
187  case 'c':
188  config.run_n3000 = 1;
189  break;
190  case 'v':
191  fpgaGetOPAECVersionString(version, sizeof(version));
192  fpgaGetOPAECBuildString(build, sizeof(build));
193  printf("hello_fpga %s %s\n",
194  version, build);
195  return -1;
196 
197  default: /* invalid option */
198  fprintf(stderr, "Invalid cmdline option \n");
199  return FPGA_EXCEPTION;
200  }
201  }
202 
203  return FPGA_OK;
204 }

206 fpga_result find_fpga(fpga_properties device_filter,
207  fpga_guid afu_guid,
208  fpga_token *accelerator_token,
209  uint32_t *num_matches_accelerators)
210 {
211  fpga_properties filter = NULL;
212  fpga_result res1;
213  fpga_result res2 = FPGA_OK;
214 
215  res1 = fpgaCloneProperties(device_filter, &filter);
216  ON_ERR_GOTO(res1, out, "cloning properties object");
217 
218  res1 = fpgaPropertiesSetObjectType(filter, FPGA_ACCELERATOR);
219  ON_ERR_GOTO(res1, out_destroy, "setting object type");
220 
221  res1 = fpgaPropertiesSetGUID(filter, afu_guid);
222  ON_ERR_GOTO(res1, out_destroy, "setting GUID");
223 
224  res1 = fpgaEnumerate(&filter, 1, accelerator_token, 1, num_matches_accelerators);
225  ON_ERR_GOTO(res1, out_destroy, "enumerating accelerators");
226 
227 out_destroy:
228  res2 = fpgaDestroyProperties(&filter);
229  ON_ERR_GOTO(res2, out, "destroying properties object");
230 out:
231  return res1 != FPGA_OK ? res1 : res2;
232 }

234 /* Is the FPGA simulated with ASE? */
235 bool probe_for_ase(void)
236 {
237  fpga_result r = FPGA_OK;
238  uint16_t device_id = 0;
239  fpga_properties filter = NULL;
240  uint32_t num_matches = 1;
241  fpga_token fme_token;
242 
243  /* Connect to the FPGA management engine */
244  fpgaGetProperties(NULL, &filter);
245  fpgaPropertiesSetObjectType(filter, FPGA_DEVICE);
246 
247  /* Connecting to one is sufficient to find ASE */
248  fpgaEnumerate(&filter, 1, &fme_token, 1, &num_matches);
249  if (0 != num_matches) {
250  /* Retrieve the device ID of the FME */
251  fpgaDestroyProperties(&filter);
252  fpgaGetProperties(fme_token, &filter);
253  r = fpgaPropertiesGetDeviceID(filter, &device_id);
254  fpgaDestroyToken(&fme_token);
255  }
256  fpgaDestroyProperties(&filter);
257 
258  /* ASE's device ID is 0xa5e */
259  return ((FPGA_OK == r) && (0xa5e == device_id));
260 }

262 fpga_result find_nlb_n3000(fpga_handle accelerator_handle,
263  uint64_t *afu_baddr)
264 {
265 
266  fpga_result res1 = FPGA_OK;
267  int end_of_list = 0;
268  int nlb0_found = 0;
269  uint64_t header = 0;
270  uint64_t uuid_hi = 0;
271  uint64_t uuid_lo = 0;
272  uint64_t next_offset = 0;
273  uint64_t nlb0_offset = 0;
274 
275  /* find NLB0 in AFU */
276  do {
277  // Read the next feature header
278  res1 = fpgaReadMMIO64(accelerator_handle, 0, nlb0_offset, &header);
279  ON_ERR_GOTO(res1, out_exit, "fpgaReadMMIO64");
280  res1 = fpgaReadMMIO64(accelerator_handle, 0, nlb0_offset+8, &uuid_lo);
281  ON_ERR_GOTO(res1, out_exit, "fpgaReadMMIO64");
282  res1 = fpgaReadMMIO64(accelerator_handle, 0, nlb0_offset+16, &uuid_hi);
283  ON_ERR_GOTO(res1, out_exit, "fpgaReadMMIO64");
284  // printf("%zx: %zx %zx %zx\n", nlb0_offset, header, uuid_lo, uuid_hi);
285  if ((((header >> 60) & 0xf) == 0x1) &&
286  (uuid_lo == FPGA_NLB0_UUID_L) && (uuid_hi == FPGA_NLB0_UUID_H)) {
287  nlb0_found = 1;
288  break;
289  }
290  // End of the list flag
291  end_of_list = (header >> 40) & 1;
292  // Move to the next feature header
293  next_offset = (header >> 16) & 0xffffff;
294  if ((next_offset == 0xffff) || (next_offset == 0)) {
295  nlb0_found = 0;
296  break;
297  }
298  nlb0_offset = nlb0_offset + next_offset;
299  } while (!end_of_list);
300 
301  if (!nlb0_found) {
302  printf("AFU NLB0 not found\n");
303  return FPGA_EXCEPTION;
304  }
305 
306  printf("AFU NLB0 found @ %zx\n", nlb0_offset);
307  *afu_baddr = nlb0_offset;
308  return FPGA_OK;
309 
310 out_exit:
311  return FPGA_EXCEPTION;
312 }

314 int main(int argc, char *argv[])
315 {
316  fpga_token accelerator_token;
317  fpga_handle accelerator_handle;
318  fpga_guid guid;
319  uint32_t num_matches_accelerators = 0;
320  uint32_t use_ase;
321 
322  volatile uint64_t *dsm_ptr = NULL;
323  volatile uint64_t *status_ptr = NULL;
324  volatile uint64_t *input_ptr = NULL;
325  volatile uint64_t *output_ptr = NULL;
326 
327  uint64_t dsm_wsid;
328  uint64_t input_wsid;
329  uint64_t output_wsid;
330  uint32_t i;
331  uint32_t timeout;
332  fpga_result res1 = FPGA_OK;
333  fpga_result res2 = FPGA_OK;
334  fpga_properties device_filter = NULL;
335 
336  res1 = fpgaGetProperties(NULL, &device_filter);
337  if (res1 != FPGA_OK) {
338  print_err("failed to allocate properties.\n", res1);
339  return 1;
340  }
341 
342  if (opae_set_properties_from_args(device_filter,
343  &res1,
344  &argc,
345  argv)) {
346  print_err("failed arg parse.\n", res1);
347  res1 = FPGA_EXCEPTION;
348  goto out_exit;
349  } else if (res1) {
350  print_err("failed to set properties.\n", res1);
351  goto out_exit;
352  }
353 
354  res1 = parse_args(argc, argv);
355  if ((int)res1 < 0)
356  goto out_exit;
357  ON_ERR_GOTO(res1, out_exit, "parsing arguments");
358 
359  if (config.run_n3000) {
360  if (uuid_parse(N3000_AFUID, guid) < 0)
361  res1 = FPGA_EXCEPTION;
362  } else {
363  if (uuid_parse(NLB0_AFUID, guid) < 0)
364  res1 = FPGA_EXCEPTION;
365  }
366 
367  ON_ERR_GOTO(res1, out_exit, "parsing guid");
368 
369  use_ase = probe_for_ase();
370  if (use_ase) {
371  printf("Running in ASE mode\n");
372  }
373 
374  /* Look for accelerator with NLB0_AFUID */
375  res1 = find_fpga(device_filter,
376  guid,
377  &accelerator_token,
378  &num_matches_accelerators);
379  ON_ERR_GOTO(res1, out_exit, "finding FPGA accelerator");
380 
381  if (num_matches_accelerators == 0) {
382  res1 = FPGA_NOT_FOUND;
383  }
384  ON_ERR_GOTO(res1, out_exit, "no matching accelerator");
385 
386  if (num_matches_accelerators > 1) {
387  printf("Found more than one suitable accelerator. ");
388  }
389 
390 
391  /* Open accelerator and map MMIO */
392  res1 = fpgaOpen(accelerator_token, &accelerator_handle, config.open_flags);
393  ON_ERR_GOTO(res1, out_destroy_tok, "opening accelerator");
394 
395  res1 = fpgaMapMMIO(accelerator_handle, 0, NULL);
396  ON_ERR_GOTO(res1, out_close, "mapping MMIO space");
397 
398 
399  /* Allocate buffers */
400  res1 = fpgaPrepareBuffer(accelerator_handle, LPBK1_DSM_SIZE,
401  (void **)&dsm_ptr, &dsm_wsid, 0);
402  ON_ERR_GOTO(res1, out_close, "allocating DSM buffer");
403 
404  res1 = fpgaPrepareBuffer(accelerator_handle, LPBK1_BUFFER_ALLOCATION_SIZE,
405  (void **)&input_ptr, &input_wsid, 0);
406  ON_ERR_GOTO(res1, out_free_dsm, "allocating input buffer");
407 
408  res1 = fpgaPrepareBuffer(accelerator_handle, LPBK1_BUFFER_ALLOCATION_SIZE,
409  (void **)&output_ptr, &output_wsid, 0);
410  ON_ERR_GOTO(res1, out_free_input, "allocating output buffer");
411 
412  printf("Running Test\n");
413 
414  /* Initialize buffers */
415  memset((void *)dsm_ptr, 0, LPBK1_DSM_SIZE);
416  memset((void *)input_ptr, 0xAF, LPBK1_BUFFER_SIZE);
417  memset((void *)output_ptr, 0xBE, LPBK1_BUFFER_SIZE);
418 
419  cache_line *cl_ptr = (cache_line *)input_ptr;
420  for (i = 0; i < LPBK1_BUFFER_SIZE / CL(1); ++i) {
421  cl_ptr[i].uint[15] = i+1; /* set the last uint in every cacheline */
422  }
423 
424 
425  /* Reset accelerator */
426  res1 = fpgaReset(accelerator_handle);
427  ON_ERR_GOTO(res1, out_free_output, "resetting accelerator");
428 
429  uint64_t nlb_base_addr = 0;
430 
431  if (config.run_n3000) {
432  res1 = find_nlb_n3000(accelerator_handle, &nlb_base_addr);
433  ON_ERR_GOTO(res1, out_free_output, "finding nlb in AFU");
434  }
435 
436  /* Program DMA addresses */
437  uint64_t iova = 0;
438  res1 = fpgaGetIOAddress(accelerator_handle, dsm_wsid, &iova);
439  ON_ERR_GOTO(res1, out_free_output, "getting DSM IOVA");
440 
441  res1 = fpgaWriteMMIO64(accelerator_handle, 0, nlb_base_addr + CSR_AFU_DSM_BASEL, iova);
442  ON_ERR_GOTO(res1, out_free_output, "writing CSR_AFU_DSM_BASEL");
443 
444  res1 = fpgaWriteMMIO32(accelerator_handle, 0, nlb_base_addr + CSR_CTL, 0);
445  ON_ERR_GOTO(res1, out_free_output, "writing CSR_CFG");
446  res1 = fpgaWriteMMIO32(accelerator_handle, 0, nlb_base_addr + CSR_CTL, 1);
447  ON_ERR_GOTO(res1, out_free_output, "writing CSR_CFG");
448 
449  res1 = fpgaGetIOAddress(accelerator_handle, input_wsid, &iova);
450  ON_ERR_GOTO(res1, out_free_output, "getting input IOVA");
451  res1 = fpgaWriteMMIO64(accelerator_handle, 0, nlb_base_addr + CSR_SRC_ADDR, CACHELINE_ALIGNED_ADDR(iova));
452  ON_ERR_GOTO(res1, out_free_output, "writing CSR_SRC_ADDR");
453 
454  res1 = fpgaGetIOAddress(accelerator_handle, output_wsid, &iova);
455  ON_ERR_GOTO(res1, out_free_output, "getting output IOVA");
456  res1 = fpgaWriteMMIO64(accelerator_handle, 0, nlb_base_addr + CSR_DST_ADDR, CACHELINE_ALIGNED_ADDR(iova));
457  ON_ERR_GOTO(res1, out_free_output, "writing CSR_DST_ADDR");
458 
459  res1 = fpgaWriteMMIO32(accelerator_handle, 0, nlb_base_addr + CSR_NUM_LINES, LPBK1_BUFFER_SIZE / CL(1));
460  ON_ERR_GOTO(res1, out_free_output, "writing CSR_NUM_LINES");
461  res1 = fpgaWriteMMIO32(accelerator_handle, 0, nlb_base_addr + CSR_CFG, 0x42000);
462  ON_ERR_GOTO(res1, out_free_output, "writing CSR_CFG");
463 
464  status_ptr = dsm_ptr + DSM_STATUS_TEST_COMPLETE/sizeof(uint64_t);
465 
466 
467  /* Start the test */
468  res1 = fpgaWriteMMIO32(accelerator_handle, 0, nlb_base_addr + CSR_CTL, 3);
469  ON_ERR_GOTO(res1, out_free_output, "writing CSR_CFG");
470 
471  /* Wait for test completion */
472  timeout = TEST_TIMEOUT;
473  while (0 == ((*status_ptr) & 0x1)) {
474  usleep(100);
475  if (!use_ase && (--timeout == 0)) {
476  res1 = FPGA_EXCEPTION;
477  ON_ERR_GOTO(res1, out_free_output, "test timed out");
478  }
479  }
480 
481  /* Stop the device */
482  res1 = fpgaWriteMMIO32(accelerator_handle, 0, nlb_base_addr + CSR_CTL, 7);
483  ON_ERR_GOTO(res1, out_free_output, "writing CSR_CFG");
484 
485  /* Wait for the AFU's read/write traffic to complete */
486  uint32_t afu_traffic_trips = 0;
487  while (afu_traffic_trips < 100) {
488  /*
489  * CSR_STATUS1 holds two 32 bit values: num pending reads and writes.
490  * Wait for it to be 0.
491  */
492  uint64_t s1;
493  res1 = fpgaReadMMIO64(accelerator_handle, 0, nlb_base_addr + CSR_STATUS1, &s1);
494  ON_ERR_GOTO(res1, out_free_output, "reading CSR_STATUS1");
495  if (s1 == 0) {
496  break;
497  }
498 
499  afu_traffic_trips += 1;
500  usleep(1000);
501  }
502 
503  /* Check output buffer contents */
504  for (i = 0; i < LPBK1_BUFFER_SIZE; i++) {
505  if (((uint8_t *)output_ptr)[i] != ((uint8_t *)input_ptr)[i]) {
506  fprintf(stderr, "Output does NOT match input "
507  "at offset %i!\n", i);
508  break;
509  }
510  }
511 
512  printf("Done Running Test\n");
513 
514 
515  /* Release buffers */
516 out_free_output:
517  res2 = fpgaReleaseBuffer(accelerator_handle, output_wsid);
518  ON_ERR_GOTO(res2, out_free_input, "releasing output buffer");
519 out_free_input:
520  res2 = fpgaReleaseBuffer(accelerator_handle, input_wsid);
521  ON_ERR_GOTO(res2, out_free_dsm, "releasing input buffer");
522 out_free_dsm:
523  res2 = fpgaReleaseBuffer(accelerator_handle, dsm_wsid);
524  ON_ERR_GOTO(res2, out_unmap, "releasing DSM buffer");
525 
526  /* Unmap MMIO space */
527 out_unmap:
528  res2 = fpgaUnmapMMIO(accelerator_handle, 0);
529  ON_ERR_GOTO(res2, out_close, "unmapping MMIO space");
530 
531  /* Release accelerator */
532 out_close:
533  res2 = fpgaClose(accelerator_handle);
534  ON_ERR_GOTO(res2, out_destroy_tok, "closing accelerator");
535 
536  /* Destroy token */
537 out_destroy_tok:
538  res2 = fpgaDestroyToken(&accelerator_token);
539  ON_ERR_GOTO(res2, out_exit, "destroying token");
540 
541 out_exit:
542  fpgaDestroyProperties(&device_filter);
543  return res1 != FPGA_OK ? res1 : res2;
544 }
