424  virtual int run(CLI::App *app, test_command::ptr_t test) override
425  {
426  int res = exit_codes::not_run;
427 
428  logger_->set_pattern(" %v");
429  // Info prints details of an individual run. Turn it on if doing only one test
430  // and the user hasn't changed level from the default.
431  if ((log_level_.compare("warning") == 0) && !he_test_all_)
432  logger_->set_level(spdlog::level::info);
433 
434  logger_->info("starting test run, count of {0:d}", count_);
435  uint32_t count = 0;
436  try {
437  while (count < count_) {
438  logger_->debug("starting iteration: {0:d}", count+1);
439 
440  res = test_afu::run(app, test);
441  count++;
442  logger_->debug("end iteration: {0:d}", count);
443  if (res)
444  break;
445  }
446  } catch(std::exception &ex) {
447  logger_->error(ex.what());
448  res = exit_codes::exception;
449  }
450 
451  auto pass = res == exit_codes::success ? "PASS" : "FAIL";
452  logger_->info("Test {}({}): {}", test->name(), count, pass);
453  spdlog::drop_all();
454  return res;
455  }
