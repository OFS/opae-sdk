107 static fpga_result fpga_open(struct n5010 *n5010)
108 {
109  fpga_result res;
110  uint32_t num = 0;
111  res = fpgaPropertiesSetObjectType(n5010->filter, FPGA_ACCELERATOR);
112  if (res != FPGA_OK) {
113  fprintf(stderr, "failed to set object type: %s\n",
114  fpgaErrStr(res));
115  goto error;
116  }
117  res = fpgaPropertiesSetGUID(n5010->filter, n5010->guid);
118  if (res != FPGA_OK) {
119  fprintf(stderr, "failed to set guid: %s\n", fpgaErrStr(res));
120  goto error;
121  }
122 
123  res = fpgaEnumerate(&n5010->filter, 1, &n5010->token, 1, &num);
124  if (res != FPGA_OK) {
125  fprintf(stderr, "failed to enumerate fpga: %s\n",
126  fpgaErrStr(res));
127  goto error;
128  }
129 
130  if (num == 0) {
131  res = FPGA_NOT_FOUND;
132  fprintf(stderr, "failed to find fpga with guid: %s\n",
133  fpgaErrStr(res));
134  goto error;
135  }
136 
137  res = fpgaOpen(n5010->token, &n5010->handle, FPGA_OPEN_SHARED);
138 
139  if (res != FPGA_OK) {
140  fprintf(stderr, "failed to open fpga: %s\n", fpgaErrStr(res));
141  goto error;
142  }
143 
144  res = fpgaMapMMIO(n5010->handle, 0, NULL);
145  if (res != FPGA_OK) {
146  fprintf(stderr, "failed to map io memory: %s\n",
147  fpgaErrStr(res));
148  fpgaClose(n5010->handle);
149  goto error;
150  }
151 
152 error:
153  return res;
154 }

180 static fpga_result fpga_base(struct n5010 *n5010)
181 {
182  fpga_result res;
183 
184  while (1) {
185  uint64_t header;
186  struct {
187  union {
188  uuid_t u128;
189  uint64_t u64[2];
190  };
191  } uuid;
192  int i;
193 
194 
195  res = fpgaReadMMIO64(n5010->handle, 0, n5010->base, &header);
196  if (res != FPGA_OK) {
197  fprintf(stderr, "failed to read dfh: %s\n",
198  fpgaErrStr(res));
199  break;
200  }
201 
202  res = fpgaReadMMIO64(n5010->handle, 0, n5010->base + 8,
203  &uuid.u64[1]);
204  if (res != FPGA_OK) {
205  fprintf(stderr, "failed to read uuid high: %s\n",
206  fpgaErrStr(res));
207  break;
208  }
209 
210  res = fpgaReadMMIO64(n5010->handle, 0, n5010->base + 16,
211  &uuid.u64[0]);
212  if (res != FPGA_OK) {
213  fprintf(stderr, "failed to read uuid low: %s\n",
214  fpgaErrStr(res));
215  break;
216  }
217 
218  for (i = 0; i < 2; i++)
219  uuid.u64[i] = be64toh(uuid.u64[i]);
220 
221  if (DFH_TYPE(header) == DFH_TYPE_AFU
222  && uuid_compare(uuid.u128, n5010->guid) == 0) {
223  }
224  break;
225 
226  if (DFH_EOL(header)) {
227  fprintf(stderr, "no matching dfh found: 0x%016jx\n",
228  header);
229  res = FPGA_NOT_FOUND;
230  break;
231  }
232 
233  if (DFH_NEXT(header) == 0 || DFH_NEXT(header) == 0xffff) {
234  fprintf(stderr, "next dfh not found: 0x%016jx\n",
235  header);
236  res = FPGA_NOT_FOUND;
237  break;
238  }
239 
240  n5010->base += DFH_NEXT(header);
241  }
242 
243  return res;
244 }

344 static bool parse_port(struct n5010 *n5010, const char *port_str)
345 {
346  char *endptr;
347  unsigned long int port = strtoul(port_str, &endptr, 0);
348  if (port_str == endptr) {
349  fprintf(stderr, "missing port number: '%s'\n", port_str);
350  return false;
351  }
352 
353  if (port > MAX_PORT) {
354  fprintf(stderr, "port number too big: '%lu'\n", port);
355  return false;
356  }
357 
358  n5010->port = port;
359  return true;
360 }

385 static int parse_args(int argc, char **argv, struct n5010 *n5010)
386 {
387  struct option options[] = {
388  {"help", no_argument, NULL, 'h'},
389  {"guid", required_argument, NULL, 'g'},
390  {"mode", required_argument, NULL, 'm'},
391  {"port", required_argument, NULL, 'p'},
392  {"debug", no_argument, NULL, 'd'},
393  {NULL, 0, NULL, 0},
394  };
395 
396  int c;
397  n5010->port = INVALID_PORT;
398 
399  while (1) {
400  c = getopt_long_only(argc, argv, "hg:m:p:d", options, NULL);
401  if (c == -1)
402  break;
403 
404  switch (c) {
405  case 'h':
406  print_usage(stdout);
407  exit(EXIT_SUCCESS);
408  case 'g':
409  if (uuid_parse(optarg, n5010->guid) < 0) {
410  fprintf(stderr, "unparsable uuid: '%s'\n",
411  optarg);
412  return EXIT_FAILURE;
413  }
414  break;
415  case 'm':
416  if (!parse_mode(n5010, optarg))
417  return EXIT_FAILURE;
418  break;
419  case 'p':
420  if (!parse_port(n5010, optarg))
421  return EXIT_FAILURE;
422  break;
423  case 'd':
424  n5010->debug = true;
425  break;
426  case '?':
427  default:
428  return EXIT_FAILURE;
429  }
430  }
431 
432  if (uuid_is_null(n5010->guid)) {
433  fprintf(stderr, "Using default guid %s\n", default_guid);
434  uuid_parse(default_guid, n5010->guid);
435  }
436 
437  return EXIT_SUCCESS;
438 }

440 int main(int argc, char **argv)
441 {
442  struct n5010 n5010 = {
443  .filter = NULL, .test = n5010_test, .debug = false};
444  fpga_result res;
445  int err;
446 
447  res = fpgaGetProperties(NULL, &n5010.filter);
448  if (res != FPGA_OK) {
449  fprintf(stderr, "failed to get properties: %s\n",
450  fpgaErrStr(res));
451  return res;
452  }
453 
454  if (opae_set_properties_from_args(n5010.filter, &res, &argc, argv)) {
455  fprintf(stderr, "failed arg parse.\n");
456  res = FPGA_EXCEPTION;
457  goto error2;
458  } else if (res) {
459  fprintf(stderr, "failed to set properties.\n");
460  goto error2;
461  }
462 
463  err = parse_args(argc, argv, &n5010);
464  if (err != 0)
465  goto error2;
466 
467  // Install Control-C handler
468  signal(SIGINT, signal_callback_handler);
469 
470  res = fpga_open(&n5010);
471  if (res != FPGA_OK)
472  goto error;
473 
474  res = fpga_base(&n5010);
475  if (res != FPGA_OK)
476  goto error;
477 
478  fpga_dump(&n5010, 0, 8);
479 
480  res = n5010.test->func(&n5010);
481  if (res != FPGA_OK)
482  goto error;
483  fpga_dump(&n5010, 0, 8);
484 
485 error:
486  fpga_close(&n5010);
487 
488 
489 error2:
490  fpgaDestroyProperties(&n5010.filter);
491 
492 
493  return res == FPGA_OK ? EXIT_SUCCESS : res;
494 }
