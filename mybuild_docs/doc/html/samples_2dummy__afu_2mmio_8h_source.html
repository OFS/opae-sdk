63 template<typename T>
64 inline void write_verify(dummy_afu *afu, uint32_t addr, T value)
65 {
66  afu->write<T>(addr, 0);
67  if (afu->read<T>(addr) != 0)
68  throw std::logic_error("mmio write failed");
69  afu->write<T>(addr, value);
70  auto rd_value = afu->read<T>(addr);
71  if (rd_value != value)
72  throw std::logic_error("mmio write failed");
73  afu->write<T>(addr, 0);
74  if (afu->read<T>(addr) != 0)
75  throw std::logic_error("mmio write failed");
76 }
