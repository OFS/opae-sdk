98 bool buffer_allocate(void** addr, uint64_t len, uint32_t numa_node)
99 {
100  void* addr_local = NULL;
101  int i = 0;
102  long status = 0;
103  unsigned long mask[4];
104  unsigned int bits_per_UL = sizeof(unsigned long) * 8;
105 
106  for (i = 0; i < 4; i++) mask[i] = 0;
107  mask[numa_node / bits_per_UL] |= 1UL << (numa_node % bits_per_UL);
108 
109  if (len > MiB(2))
110  addr_local = mmap(ADDR, len, PROTECTION, FLAGS_1G, 0, 0);
111  else if (len > KiB(4))
112  addr_local = mmap(ADDR, len, PROTECTION, FLAGS_2M, 0, 0);
113  else
114  addr_local = mmap(ADDR, len, PROTECTION, FLAGS_4K, 0, 0);
115 
116  if (addr_local == MAP_FAILED) {
117  if (errno == ENOMEM) {
118  if (len > MiB(2))
119  cerr <<"Could not allocate buffer (no free 1 "
120  "GiB huge pages)";
121  if (len > KiB(4))
122  cerr << "Could not allocate buffer (no free 2 "
123  "MiB huge pages)";
124  else
125  cerr <<"Could not allocate buffer (out of "
126  "memory)";
127  return false;
128  }
129  cerr << "CXL cache mmap failed:"<< strerror(errno) << endl;
130  return false;
131  }
132 
133  if (addr_local == NULL) {
134  cerr << "Unable to mmap" << endl;
135  return false;
136  }
137 
138  status = syscall(__NR_mbind, addr_local, len, 2, &mask, numa_node + 2, 1);
139  if (status != 0) {
140  cerr << "buffer_allocate(): unable to mbind:"
141  << strerror(errno) << endl;
142  return false;
143  }
144 
145  *addr = addr_local;
146  return true;
147 }

186 union pcie_address {
187  struct {
188  uint32_t function : 3;
189  uint32_t device : 5;
190  uint32_t bus : 8;
191  uint32_t domain : 16;
192  } fields;
193  uint32_t value;
194 
195  static pcie_address parse(const char *s) {
196  auto deleter = [&](regex_t *r) {
197  regfree(r);
198  delete r;
199  };
200  std::unique_ptr<regex_t, decltype(deleter)> re(new regex_t, deleter);
201  regmatch_t matches[MATCHES_SIZE];
202 
203  int reg_res = regcomp(re.get(), sbdf_pattern, REG_EXTENDED | REG_ICASE);
204  if (reg_res)
205  throw std::runtime_error("could not compile regex");
206 
207  reg_res = regexec(re.get(), s, MATCHES_SIZE, matches, 0);
208  if (reg_res)
209  throw std::runtime_error("pcie address not valid format");
210 
211  uint16_t domain, bus, device, function;
212  if (!parse_match_int(s, matches[2], domain, 16))
213  domain = 0;
214  if (!parse_match_int(s, matches[3], bus, 16))
215  throw std::runtime_error("error parsing pcie address");
216  if (!parse_match_int(s, matches[4], device, 16))
217  throw std::runtime_error("error parsing pcie address");
218  if (!parse_match_int(s, matches[5], function))
219  throw std::runtime_error("error parsing; pcie address");
220  pcie_address a;
221  a.fields.domain = domain;
222  a.fields.bus = bus;
223  a.fields.device = device;
224  a.fields.function = function;
225  return a;
226  }

306  int find_dev_feature() {
307  glob_t pglob;
308  glob_t dev_pglob;
309  char feature_path[MAX_SIZE] = {0};
310  int gres = 0;
311  uint64_t value = 0;
312  size_t i = 0;
313  size_t dev_index = 0;
314 
315  if (!pci_addr_.empty()) {
316  if (snprintf(feature_path, sizeof(feature_path), FEATURE_DEV,
317  pci_addr_.c_str()) < 0) {
318  cerr << "snprintf buffer overflow" << endl;
319  return 1;
320  }
321  } else {
322  if (snprintf(feature_path, sizeof(feature_path), FEATURE_DEV, "*:*:*.*") <
323  0) {
324  cerr << "snprintf buffer overflow" << endl;
325  return 2;
326  }
327  }
328 
329  gres = glob(feature_path, 0, NULL, &pglob);
330  if (gres) {
331  cerr << "Failed pattern match" << feature_path << ":" << strerror(errno)
332  << endl;
333  globfree(&pglob);
334  return 3;
335  }
336 
337  for (i = 0; i < pglob.gl_pathc; i++) {
338  bool retval = sysfs_read_u64(pglob.gl_pathv[i], &value);
339  if (!retval) {
340  cerr << "Failed to read sysfs value" << endl;
341  continue;
342  }
343 
344  if (current_command()->featureid() == value) {
345  string str(pglob.gl_pathv[i]);
346  string substr_dev(str.substr(0, str.rfind("/")));
347 
348  substr_dev.append("/dfl-cxl-cache/dfl-cxl-cache*");
349  gres = glob(substr_dev.c_str(), GLOB_NOSORT, NULL, &dev_pglob);
350  if (gres) {
351  cerr << "Failed pattern match" << substr_dev.c_str() << ":"
352  << strerror(errno) << endl;
353  globfree(&dev_pglob);
354  return 4;
355  }
356 
357  string str1(dev_pglob.gl_pathv[0]);
358  globfree(&dev_pglob);
359  dev_path_[dev_index].append("/dev");
360  dev_path_[dev_index].append(str1.substr(str1.rfind("/"), 16));
361  dev_index++;
362  }
363  }
364 
365  if (pglob.gl_pathv) {
366  globfree(&pglob);
367  }
368 
369  if (dev_index > 0) {
370  return 0;
371  }
372 
373  return 5;
374  }
