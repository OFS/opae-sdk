125 static fpga_result fpga_open(struct n5010 *n5010)
126 {
127  fpga_result res;
128  uint32_t num = 0;
129 
130  res = fpgaPropertiesSetObjectType(n5010->filter, FPGA_ACCELERATOR);
131  if (res != FPGA_OK) {
132  fprintf(stderr, "failed to set object type: %s\n", fpgaErrStr(res));
133  return res;
134  }
135 
136  res = fpgaPropertiesSetGUID(n5010->filter, n5010->guid);
137  if (res != FPGA_OK) {
138  fprintf(stderr, "failed to set guid: %s\n", fpgaErrStr(res));
139  return res;
140  }
141 
142  res = fpgaEnumerate(&n5010->filter, 1, &n5010->token, 1, &num);
143  if (res != FPGA_OK) {
144  fprintf(stderr, "failed to enumerate fpga: %s\n", fpgaErrStr(res));
145  return res;
146  }
147 
148  if (num == 0) {
149  res = FPGA_NOT_FOUND;
150  fprintf(stderr, "failed to find fpga with guid: %s\n", fpgaErrStr(res));
151  return res;
152  }
153 
154  res = fpgaOpen(n5010->token, &n5010->handle, n5010->open_mode);
155  if (res != FPGA_OK) {
156  fprintf(stderr, "failed to open fpga: %s\n", fpgaErrStr(res));
157  return res;
158  }
159 
160  res = fpgaMapMMIO(n5010->handle, 0, NULL);
161  if (res != FPGA_OK) {
162  fprintf(stderr, "failed to map io memory: %s\n", fpgaErrStr(res));
163  return res;
164  }
165 
166  res = fpgaReset(n5010->handle);
167  if (res != FPGA_OK) {
168  fprintf(stderr, "failed to reset fpga: %s\n", fpgaErrStr(res));
169  return res;
170  }
171 
172  return res;
173 }

202 static fpga_result fpga_base(struct n5010 *n5010)
203 {
204  fpga_result res;
205 
206  while (1) {
207  uint64_t header;
208  struct {
209  union {
210  uuid_t u128;
211  uint64_t u64[2];
212  };
213  } uuid;
214  int i;
215 
216  res = fpgaReadMMIO64(n5010->handle, 0, n5010->base, &header);
217  if (res != FPGA_OK) {
218  fprintf(stderr, "failed to read dfh: %s\n", fpgaErrStr(res));
219  break;
220  }
221 
222  res = fpgaReadMMIO64(n5010->handle, 0, n5010->base + 8, &uuid.u64[1]);
223  if (res != FPGA_OK) {
224  fprintf(stderr, "failed to read uuid high: %s\n", fpgaErrStr(res));
225  break;
226  }
227 
228  res = fpgaReadMMIO64(n5010->handle, 0, n5010->base + 16, &uuid.u64[0]);
229  if (res != FPGA_OK) {
230  fprintf(stderr, "failed to read uuid low: %s\n", fpgaErrStr(res));
231  break;
232  }
233 
234  for (i = 0; i < 2; i++)
235  uuid.u64[i] = be64toh(uuid.u64[i]);
236 
237  if (DFH_TYPE(header) == DFH_TYPE_AFU && uuid_compare(uuid.u128, n5010->guid) == 0)
238  break;
239 
240  if (DFH_EOL(header)) {
241  fprintf(stderr, "no matching dfh found: 0x%016jx\n", header);
242  res = FPGA_NOT_FOUND;
243  break;
244  }
245 
246  if (DFH_NEXT(header) == 0 || DFH_NEXT(header) == 0xffff) {
247  fprintf(stderr, "next dfh not found: 0x%016jx\n", header);
248  res = FPGA_NOT_FOUND;
249  break;
250  }
251 
252  n5010->base += DFH_NEXT(header);
253  }
254 
255  return res;
256 }

288 static fpga_result fpga_start(struct n5010 *n5010, uint64_t offset, uint64_t num_banks)
289 {
290  uint64_t ctrl;
291  fpga_result res = FPGA_OK;
292 
293  ctrl = ((uint64_t)1 << num_banks) - 1;
294  printf("starting PRBS generators\n");
295 
296  fpga_dump(n5010, offset, 1);
297 
298  res = fpgaWriteMMIO64(n5010->handle, 0, n5010->base + offset, ctrl);
299  if (res != FPGA_OK) {
300  fprintf(stderr, "failed to start PRBS generators: %s\n", fpgaErrStr(res));
301  goto error;
302  }
303 
304  usleep(1000000);
305 
306  fpga_dump(n5010, offset, 1);
307 
308 error:
309  return res;

334 static fpga_result fpga_wait(struct n5010 *n5010, uint64_t offset, uint64_t init, uint64_t result)
335 {
336  struct timespec ts = {.tv_sec = 0, .tv_nsec = 100000000};
337  uint64_t status = init;
338  fpga_result res;
339 
340  printf("waiting for test to complete...\n");
341 
342  while (status == init) {
343  nanosleep(&ts, NULL);
344 
345  res = fpgaReadMMIO64(n5010->handle, 0, n5010->base + offset, &status);
346  if (res != FPGA_OK) {
347  fprintf(stderr, "failed to read status: %s\n", fpgaErrStr(res));
348  goto error;
349  }
350 
351  if (n5010->debug)
352  printf("reg: 0x%04jx (DDR status), val: 0x%016jx\n", offset, status);
353  }
354 
355  if (status != result) {
356  fprintf(stderr, "failed with unexpected result: expected: 0x%016jx, got: 0x%016jx\n", result, status);
357  res = FPGA_EXCEPTION;
358  goto error;
359  }
360 
361 error:
362  return res;
363 }

365 static fpga_result fpga_test_ddr_prbs(struct n5010 *n5010)
366 {
367  uint64_t num_banks, stat, i;
368  fpga_result res;
369 
370  printf("starting DDR PRBS read/write test\n");
371 
372  fpga_dump(n5010, DDR_TEST_MODE1_CTRL, 6);
373 
374  // Read number of num_banks from stat
375  res = fpga_banks(n5010, DDR_TEST_MODE1_STAT, &num_banks);
376  if (res != FPGA_OK)
377  goto error;
378 
379  // Clear PRBS start bits
380  res = fpga_stop(n5010, DDR_TEST_MODE1_CTRL);
381  if (res != FPGA_OK)
382  goto error;
383 
384  // Expect DDR_TEST_MODE1_BANK_STAT(bank) to return 0 while PRBS not running
385  for (i = 0; i < num_banks; i++) {
386  res = fpgaReadMMIO64(n5010->handle, 0, n5010->base + DDR_TEST_MODE1_BANK_STAT(i), &stat);
387  if (res != FPGA_OK) {
388  fprintf(stderr, "failed to read DDR_TEST_MODE1_BANK_STAT(%ju): %s\n", i, fpgaErrStr(res));
389  goto error;
390  }
391 
392  if (stat != 0x0) {
393  fprintf(stderr, "Error: PRBS stat non-zero while generator idle: 0x%016jx\n", stat);
394  res = FPGA_EXCEPTION;
395  goto error;
396  }
397  }
398 
399  // Set PRBS start bits for each bank
400  res = fpga_start(n5010, DDR_TEST_MODE1_CTRL, num_banks);
401  if (res != FPGA_OK)
402  goto error;
403 
404  // Wait for PRBS pass
405  for (i = 0; i < num_banks; i++) {
406  res = fpga_wait(n5010, DDR_TEST_MODE1_BANK_STAT(i), 0, 0x39);
407  if (res != FPGA_OK)
408  goto error;
409  }
410 
411 error:
412  return res;
413 }

415 static fpga_result fpga_test_ddr_directed(struct n5010 *n5010)
416 {
417  uint64_t num_banks, stat;
418  fpga_result res;
419 
420  printf("starting DDR directed read/write test\n");
421 
422  fpga_dump(n5010, DDR_TEST_MODE0_CTRL, 4);
423 
424  // Read number of num_banks from stat
425  res = fpga_banks(n5010, DDR_TEST_MODE0_STAT, &num_banks);
426  if (res != FPGA_OK)
427  goto error;
428 
429  // Clear test generator start bits
430  res = fpga_stop(n5010, DDR_TEST_MODE0_CTRL);
431  if (res != FPGA_OK)
432  goto error;
433 
434  // Expect error status to return 0 while test generator is not running
435  res = fpgaReadMMIO64(n5010->handle, 0, n5010->base + DDR_TEST_MODE0_STAT, &stat);
436  if (res != FPGA_OK) {
437  fprintf(stderr, "Error: failed to read DDR_TEST_MODE0_STAT: %s\n", fpgaErrStr(res));
438  goto error;
439  }
440  // Check test error bits [23:16]
441  if ((stat & 0xFF0000) != 0x0) {
442  fprintf(stderr, "Error: test generator error status non-zero while generator idle: 0x%016jx\n", stat);
443  res = FPGA_EXCEPTION;
444  goto error;
445  }
446 
447  // Set start bits for each bank
448  res = fpga_start(n5010, DDR_TEST_MODE0_CTRL, num_banks);
449  if (res != FPGA_OK)
450  goto error;
451 
452  // Wait for test generator pass bits to indicate done for all banks:
453  // DDR_TEST_MODE0_STAT[23:16] : test error = 0x0
454  // DDR_TEST_MODE0_STAT[15:8] : test done = (1 << num_banks)-1
455  // DDR_TEST_MODE0_STAT[7:0] : # banks = num_banks
456  res = fpga_wait(n5010, DDR_TEST_MODE0_STAT, num_banks, (((1 << num_banks)-1) << 8) + num_banks);
457  if (res != FPGA_OK)
458  goto error;
459 
460  // Check for reported errors (DDR_TEST_MODE0_STAT[23:16])
461  res = fpgaReadMMIO64(n5010->handle, 0, n5010->base + DDR_TEST_MODE0_STAT, &stat);
462  if (res != FPGA_OK) {
463  fprintf(stderr, "Error: failed to read DDR_TEST_MODE0_STAT: %s\n", fpgaErrStr(res));
464  goto error;
465  }
466  if ((stat & 0xFF0000) != 0x0) {
467  fprintf(stderr, "Error: Test failed with the following status: 0x%016jx\n", stat);
468  res = FPGA_EXCEPTION;
469  goto error;
470  }
471 
472 error:
473  return res;
474 }

476 static fpga_result fpga_test_esram(struct n5010 *n5010)
477 {
478  struct timespec ts = {.tv_sec = 0, .tv_nsec = 100000000};
479  uint64_t stat = 0;
480  uint16_t pass = 0;
481  uint16_t fail = 0;
482  uint16_t timeout = 0;
483  fpga_result res ;
484  uint32_t polls = 0;
485 
486  printf("starting eSRAM read/write test\n");
487 
488  res = fpgaReadMMIO64(n5010->handle, 0, n5010->base + ESRAM_TEST_STAT, &stat);
489  if (res != FPGA_OK || stat != 0) {
490  fprintf(stderr, "FPGA not ready for test, status: 0x%016jx\n", stat);
491  if (res == FPGA_OK)
492  res = FPGA_EXCEPTION;
493  goto error;
494  }
495 
496  res = fpgaWriteMMIO64(n5010->handle, 0, n5010->base + ESRAM_TEST_CTRL, 1);
497  if (res != FPGA_OK) {
498  fprintf(stderr, "failed to start test: %s\n", fpgaErrStr(res));
499  goto error;
500  }
501  printf("waiting for test to complete...\n");
502 
503  do {
504  ++polls;
505  res = fpgaReadMMIO64(n5010->handle, 0, n5010->base + ESRAM_TEST_STAT, &stat);
506  if (res != FPGA_OK) {
507  fprintf(stderr, "failed to read stat: %s\n", fpgaErrStr(res));
508  goto error;
509  }
510  pass = ESRAM_TEST_PASS(stat);
511  fail = ESRAM_TEST_FAIL(stat);
512  timeout = ESRAM_TEST_TIMEOUT(stat);
513 
514 
515  if (n5010->debug) {
516  printf("pass (16x eSRAM channels) : 0x%04x\n", pass);
517  printf("fail (16x eSRAM channels) : 0x%04x\n", fail);
518  printf("timeout (16x eSRAM channels): 0x%04x\n", timeout);
519  }
520  nanosleep(&ts, NULL);
521 
522  } while (((pass | fail | timeout) != ESRAM_FM) && polls < MAX_POLLS);
523 
524  if (polls == MAX_POLLS) {
525  fprintf(stderr, "Error: Test failed FPGA not returning result within time\n");
526  res = FPGA_EXCEPTION;
527  }
528 
529  if (fail != 0x0) {
530  fprintf(stderr, "Error: Test failed on the following channels: 0x%04x\n", fail);
531  res = FPGA_EXCEPTION;
532  }
533 
534  if (timeout != 0x0) {
535  fprintf(stderr, "Error: Test timed out on the following channels: 0x%04x\n", timeout);
536  res = FPGA_EXCEPTION;
537  }
538 
539  if (pass != ESRAM_FM) {
540  fprintf(stderr, "Error: Test did not pass on all channels: 0x%04x\n", pass);
541  res = FPGA_EXCEPTION;
542  }
543 
544  if (res != FPGA_OK)
545  goto error;
546 
547 error:
548  return res;
549 }
