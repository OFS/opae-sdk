188  virtual int run(CLI::App *app, test_command::ptr_t test) override
189  {
190  int res = exit_codes::not_run;
191  logger_->info("starting test run, count of {0:d}", count_);
192  uint32_t count = 0;
193  try {
194  while (count < count_) {
195  logger_->debug("starting iteration: {0:d}", count+1);
196  res = test_afu::run(app, test);
197  count++;
198  logger_->debug("end iteration: {0:d}", count);
199  if (res)
200  break;
201  }
202  } catch(std::exception &ex) {
203  logger_->error(ex.what());
204  res = exit_codes::exception;
205  }
206  auto pass = res == exit_codes::success ? "PASS" : "FAIL";
207  logger_->info("Test {}({}): {}", test->name(), count, pass);
208  spdlog::drop_all();
209  return res;
210  }
