184 public:
185  dummy_afu(const char *afu_id = "91c2a3a1-4a23-4e21-a7cd-2b36dbf2ed73")
186  : test_afu("dummy_afu", afu_id)

192  virtual int run(CLI::App *app, test_command::ptr_t test) override
193  {
194  int res = exit_codes::not_run;
195  logger_->set_level(spdlog::level::trace);
196  logger_->info("starting test run, count of {0:d}", count_);
197  uint32_t count = 0;
198  try {
199  while (count < count_) {
200  logger_->debug("starting iteration: {0:d}", count+1);
201  handle_->reset();
202  res = test_afu::run(app, test);
203  count++;
204  logger_->debug("end iteration: {0:d}", count);
205  if (res)
206  break;
207  }
208  } catch(std::exception &ex) {
209  logger_->error(ex.what());
210  res = exit_codes::exception;
211  }
212  handle_->reset();
213  auto pass = res == exit_codes::success ? "PASS" : "FAIL";
214  logger_->info("Test {}({}): {}", test->name(), count, pass);
215  spdlog::drop_all();
216  return res;
217  }
