95 {
96 
97  UNUSED_PARAM(argc);
98  UNUSED_PARAM(argv);

100  fpga_properties filter = NULL;
101  fpga_token afc_token;
102  fpga_handle afc_handle;
103  fpga_guid guid;
104  uint32_t num_matches;
105 
106  fpga_result res = FPGA_OK;
107 
108  if (uuid_parse(HELLO_AFU_ID, guid) < 0) {
109  fprintf(stderr, "Error parsing guid '%s'\n", HELLO_AFU_ID);
110  goto out_exit;
111  }
112 
113  /* Look for AFC with MY_AFC_ID */
114  res = fpgaGetProperties(NULL, &filter);
115  ON_ERR_GOTO(res, out_exit, "creating properties object");
116 
117  res = fpgaPropertiesSetObjectType(filter, FPGA_ACCELERATOR);
118  ON_ERR_GOTO(res, out_destroy_prop, "setting object type");
119 
120  res = fpgaPropertiesSetGUID(filter, guid);
121  ON_ERR_GOTO(res, out_destroy_prop, "setting GUID");
122 
123  /* TODO: Add selection via BDF / device ID */
124 
125  res = fpgaEnumerate(&filter, 1, &afc_token, 1, &num_matches);
126  ON_ERR_GOTO(res, out_destroy_prop, "enumerating AFCs");
127 
128  if (num_matches < 1) {
129  fprintf(stderr, "AFC not found.\n");
130  res = fpgaDestroyProperties(&filter);
131  return FPGA_INVALID_PARAM;
132  }
133 
134  /* Open AFC and map MMIO */
135  res = fpgaOpen(afc_token, &afc_handle, 0);
136  ON_ERR_GOTO(res, out_destroy_tok, "opening AFC");
137 
138  res = fpgaMapMMIO(afc_handle, 0, NULL);
139  ON_ERR_GOTO(res, out_close, "mapping MMIO space");
140 
141  printf("Running Test\n");
142 
143  /* Reset AFC */
144  res = fpgaReset(afc_handle);
145  ON_ERR_GOTO(res, out_close, "resetting AFC");
146 
147  // Access mandatory AFU registers
148  uint64_t data = 0;
149  res = fpgaReadMMIO64(afc_handle, 0, AFU_DFH_REG, &data);
150  ON_ERR_GOTO(res, out_close, "reading from MMIO");
151  printf("AFU DFH REG = %08lx\n", data);
152 
153  res = fpgaReadMMIO64(afc_handle, 0, AFU_ID_LO, &data);
154  ON_ERR_GOTO(res, out_close, "reading from MMIO");
155  printf("AFU ID LO = %08lx\n", data);
156 
157  res = fpgaReadMMIO64(afc_handle, 0, AFU_ID_HI, &data);
158  ON_ERR_GOTO(res, out_close, "reading from MMIO");
159  printf("AFU ID HI = %08lx\n", data);
160 
161  res = fpgaReadMMIO64(afc_handle, 0, AFU_NEXT, &data);
162  ON_ERR_GOTO(res, out_close, "reading from MMIO");
163  printf("AFU NEXT = %08lx\n", data);
164 
165  res = fpgaReadMMIO64(afc_handle, 0, AFU_RESERVED, &data);
166  ON_ERR_GOTO(res, out_close, "reading from MMIO");
167  printf("AFU RESERVED = %08lx\n", data);
168 
169  // Access AFU user scratch-pad register
170  res = fpgaReadMMIO64(afc_handle, 0, SCRATCH_REG, &data);
171  ON_ERR_GOTO(res, out_close, "reading from MMIO");
172  printf("Reading Scratch Register (Byte Offset=%08x) = %08lx\n", SCRATCH_REG, data);
173 
174  printf("MMIO Write to Scratch Register (Byte Offset=%08x) = %08lx\n", SCRATCH_REG, SCRATCH_VALUE);
175  res = fpgaWriteMMIO64(afc_handle, 0, SCRATCH_REG, SCRATCH_VALUE);
176  ON_ERR_GOTO(res, out_close, "writing to MMIO");
177 
178  res = fpgaReadMMIO64(afc_handle, 0, SCRATCH_REG, &data);
179  ON_ERR_GOTO(res, out_close, "reading from MMIO");
180  printf("Reading Scratch Register (Byte Offset=%08x) = %08lx\n", SCRATCH_REG, data);
181  ASSERT_GOTO(data == SCRATCH_VALUE, out_close, "MMIO mismatched expected result");
182 
183  // Set Scratch Register to 0
184  printf("Setting Scratch Register (Byte Offset=%08x) = %08x\n", SCRATCH_REG, SCRATCH_RESET);
185  res = fpgaWriteMMIO64(afc_handle, 0, SCRATCH_REG, SCRATCH_RESET);
186  ON_ERR_GOTO(res, out_close, "writing to MMIO");
187  res = fpgaReadMMIO64(afc_handle, 0, SCRATCH_REG, &data);
188  ON_ERR_GOTO(res, out_close, "reading from MMIO");
189  printf("Reading Scratch Register (Byte Offset=%08x) = %08lx\n", SCRATCH_REG, data);
190  ASSERT_GOTO(data == SCRATCH_RESET, out_close, "MMIO mismatched expected result");
191 
192  printf("Done Running Test\n");
193 
194  /* Unmap MMIO space */
195  res = fpgaUnmapMMIO(afc_handle, 0);
196  ON_ERR_GOTO(res, out_close, "unmapping MMIO space");
197 
198  /* Release accelerator */
199 out_close:
200  res = fpgaClose(afc_handle);
201  ON_ERR_GOTO(res, out_destroy_tok, "closing AFC");
202 
203  /* Destroy token */
204 out_destroy_tok:
205 #ifndef USE_ASE
206  res = fpgaDestroyToken(&afc_token);
207  ON_ERR_GOTO(res, out_destroy_prop, "destroying token");
208 #endif
209 
210  /* Destroy properties object */
211 out_destroy_prop:
212  res = fpgaDestroyProperties(&filter);
213  ON_ERR_GOTO(res, out_exit, "destroying properties object");
214 
215 out_exit:
216  if(s_error_count > 0)
217  printf("Test FAILED!\n");
218 
219  return s_error_count;
220 
221 }
