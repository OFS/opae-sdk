56  double clock_freq_for(test_afu *afu) const
57  {
58  auto afu_props = afu->afu_properties();
59  auto filter = fpga::properties::get();
60 
61  // To get the FME matching our accelerator, we filter
62  // for the ssss:bb:dd and for FPGA_DEVICE object type.
63  filter->segment = afu_props->segment;
64  filter->bus = afu_props->bus;
65  filter->device = afu_props->device;
66  filter->type = FPGA_DEVICE;
67 
68  std::vector<fpga::token::ptr_t> toks =
69  fpga::token::enumerate({filter});
70 
71  if (toks.empty()) {
72  std::cerr << "Couldn't find FME for AFU!" << std::endl;
73  return INVALID_CLOCK_FREQ;
74  }
75 
76  auto fme_props = fpga::properties::get(toks[0]);
77 
78  uint32_t major = FPGA_BBS_VER_MAJOR(fme_props->bbs_id);
79 
80  if (major == BITSVER_MAJOR_S10)
81  return USER_CLKFREQ_S10;
82  else if (major == BITSVER_MAJOR_N6000)
83  return USER_CLKFREQ_N6000;
84 
85  return INVALID_CLOCK_FREQ;
86  }

88  int set_pkt_filt_dest(const std::string &dfl_dev,
89  uint64_t bin_dest_addr) const
90  {
91  struct opae_uio uio;
92 
93  if (opae_uio_open(&uio, dfl_dev.c_str())) {
94  std::cerr << "Failed to open: " << dfl_dev << std::endl;
95  return test_afu::error;
96  }
97 
98  uint8_t *mmio = nullptr;
99 
100  if (opae_uio_region_get(&uio, 0, &mmio, nullptr)) {
101  std::cerr << "Failed to get region 0 from: " << dfl_dev << std::endl;
102  opae_uio_close(&uio);
103  return test_afu::error;
104  }
105 
106  *(volatile uint64_t *)(mmio + PKT_FILT_CSR_DEST_ADDR) = bin_dest_addr;
107  uint64_t dest_addr = *(volatile uint64_t *)(mmio + PKT_FILT_CSR_DEST_ADDR);
108 
109  if (bin_dest_addr != dest_addr) {
110  std::cerr << "Packet Filter dest address mismatch! wrote: " << int_to_hex(bin_dest_addr)
111  << " read: " << int_to_hex(dest_addr) << std::endl;
112  opae_uio_close(&uio);
113  return test_afu::error;
114  }
115 
116  opae_uio_close(&uio);
117 
118  return test_afu::success;
119  }
