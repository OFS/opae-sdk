201  void read_performance(perf_data *perf, hssi_afu *hafu) const
202  {
203  perf->tx_count = data_64(hafu->mbox_read(CSR_STATS_TX_CNT_LO), hafu->mbox_read(CSR_STATS_TX_CNT_HI));
204  perf->rx_count = data_64(hafu->mbox_read(CSR_STATS_RX_CNT_LO), hafu->mbox_read(CSR_STATS_RX_CNT_HI));
205  perf->rx_good_packet_count = data_64(hafu->mbox_read(CSR_STATS_RX_GD_CNT_LO), hafu->mbox_read(CSR_STATS_RX_GD_CNT_HI));
206  perf->rx_pkt_sec = data_64(hafu->mbox_read(CSR_RX_END_TIMESTAMP_LO), hafu->mbox_read(CSR_RX_END_TIMESTAMP_HI));
207  }

209  void calc_performance(perf_data *old_perf, perf_data *new_perf, perf_data *perf, uint64_t size) const
210  {
211  perf->tx_count = new_perf->tx_count;
212  perf->rx_count = new_perf->rx_count;
213  perf->rx_good_packet_count = new_perf->rx_good_packet_count;
214  perf->rx_pkt_sec = total_per_second((new_perf->rx_count - old_perf->rx_count), (new_perf->rx_pkt_sec - old_perf->rx_pkt_sec));
215  perf->bw = (perf->rx_pkt_sec * size) >> 27;
216  }

218  uint8_t stall_enable(hssi_afu *hafu) const
219  {
220  volatile uint32_t reg;
221  reg = hafu->mbox_read(CSR_STATS_CTRL);
222  reg |= 1 << 1;
223  hafu->mbox_write(CSR_STATS_CTRL, reg);
224  return 1;
225  }

227  uint8_t stall_disable(hssi_afu *hafu) const
228  {
229  volatile uint32_t reg;
230  reg = hafu->mbox_read(CSR_STATS_CTRL);
231  reg &= ~(1 << 1);
232  hafu->mbox_write(CSR_STATS_CTRL, reg);

234  }

309  std::ostream & run_monitor(std::ostream &os, perf_data *old_perf_data, int port, hssi_afu *hafu, uint32_t timer)
310  {
311  perf_data *cur_perf_data = (perf_data*)malloc(sizeof(perf_data));
312  perf_data *new_perf_data = (perf_data*)malloc(sizeof(perf_data));
313 
314  memset(cur_perf_data, 0, sizeof(perf_data));
315 
316  capture_perf(new_perf_data, hafu);
317  calc_performance(old_perf_data, new_perf_data, cur_perf_data, start_size_);
318 
319  print_monitor_data(os, cur_perf_data, port, timer);
320  memcpy(old_perf_data, new_perf_data, sizeof(perf_data));
321 
322  return os;
323  }

368  virtual int run(test_afu *afu, CLI::App *app) override
369  {
370  (void)app;
371 
372  if (port_.size() > MAX_PORT)
373  {
374  std::cerr<< "more than " << MAX_PORT << " ports are not supported"<<std::endl;
375  return test_afu::error;
376  }
377 
378  if (port_.empty())
379  port_.push_back(0);
380 
381  hssi_afu *hafu = dynamic_cast<hssi_afu *>(afu);
382 
383  uint64_t bin_src_addr = mac_bits_for(src_addr_);
384  if (bin_src_addr == INVALID_MAC) {
385  std::cerr << "invalid MAC address: " << src_addr_ << std::endl;
386  return test_afu::error;
387  }
388 
389  uint64_t bin_dest_addr = mac_bits_for(dest_addr_);
390  if (bin_dest_addr == INVALID_MAC) {
391  std::cerr << "invalid MAC address: " << dest_addr_ << std::endl;
392  return test_afu::error;
393  }
394 
395  std::string eth_ifc = eth_ifc_;
396  if (eth_ifc == "none")
397  eth_ifc = hafu->ethernet_interface();
398 
399  std::cout << "100G loopback test" << std::endl
400  << " port: " << port_[0] << std::endl
401  << " eth_loopback: " << eth_loopback_ << std::endl
402  << " num_packets: " << num_packets_ << std::endl
403  << " gap: " << gap_ << std::endl
404  << " src_address: " << src_addr_ << std::endl
405  << " (bits): 0x" << std::hex << bin_src_addr << std::endl
406  << " dest_address: " << dest_addr_ << std::endl
407  << " (bits): 0x" << std::hex << bin_dest_addr << std::endl
408  << " pattern: " << pattern_ << std::endl
409  << " start size: " << start_size_ << std::endl
410  << " end size: " << end_size_ << std::endl
411  << " end select: " << end_select_ << std::endl
412  << " eth: " << eth_ifc << std::endl
413  << " continuous mode: " << continuous_ << std::endl
414  << " monitor duration: " << std::dec << contmonitor_ << " sec" << std::endl
415  << std::endl;
416 
417  if (eth_ifc == "") {
418  std::cout << "No eth interface, so not "
419  "honoring --eth-loopback." << std::endl;
420  } else {
421  if (eth_loopback_ == "on")
422  enable_eth_loopback(eth_ifc, true);
423  else
424  enable_eth_loopback(eth_ifc, false);
425  }
426  ctrl_config config_data;
427 
428  DFH dfh;
429  dfh.csr = hafu->read64(ETH_AFU_DFH);
430 
431  hafu->write64(TRAFFIC_CTRL_PORT_SEL, port_[0]);
432  hafu->mbox_write(CSR_CTRL1, STOP_BITS);
433 
434  uint32_t reg;
435 
436  reg = start_size_ | (end_size_ << 16);
437  hafu->mbox_write(CSR_PKT_SIZE, reg);
438  config_data.pkt_size_data = reg;
439 
440  reg = num_packets_;
441  if (dfh.major_rev < 2){
442  reg |= DATA_CNF_FIXED_MODE;
443  }
444  else {
445  if(continuous_ == "on")
446  reg = DATA_CNF_CONTINUOUS_MODE;
447  if(continuous_ == "off")
448  reg |= DATA_CNF_FIXED_MODE;
449  }
450 
451  hafu->mbox_write(CSR_CTRL0, reg);
452  config_data.ctrl0_data = reg;
453 
454  hafu->mbox_write(CSR_SRC_ADDR_LO, static_cast<uint32_t>(bin_src_addr));
455  hafu->mbox_write(CSR_SRC_ADDR_HI, static_cast<uint32_t>(bin_src_addr >> 32));
456 
457  hafu->mbox_write(CSR_DST_ADDR_LO, static_cast<uint32_t>(bin_dest_addr));
458  hafu->mbox_write(CSR_DST_ADDR_HI, static_cast<uint32_t>(bin_dest_addr >> 32));
459 
460  reg = 0;
461  if (dfh.major_rev < 2){
462  if (gap_ == "random")
463  reg |= 1 << 7;
464 
465  if (end_select_ == "pkt_num")
466  reg |= 1 << 6;
467  }
468  else{
469  if (gap_ == "random")
470  reg |= 1 << 6;
471  }
472 
473  if (pattern_ == "fixed")
474  reg |= 1 << 4;
475  else if (pattern_ == "increment")
476  reg |= 2 << 4;
477 
478  if (eth_loopback_ == "off")
479  reg |= 1 << 3;
480 
481  hafu->mbox_write(CSR_CTRL1, reg);
482  config_data.ctrl1_data = reg;
483 
484  if(port_.size() > 1) // Supporting for 2nd port register setting
485  select_port(port_[1], config_data, hafu);
486 
487  volatile uint32_t count;
488  const uint64_t interval = 100ULL;
489  do
490  {
491  count = hafu->mbox_read(CSR_TX_COUNT);
492  if (count == 0) // Added to eliminate infinite loop occured when CSR_TX_COUNT is 0
493  break;
494 
495  if (!running()) {
496  hafu->mbox_write(CSR_CTRL1, STOP_BITS);
497  return test_afu::error;
498  }
499 
500  std::this_thread::sleep_for(std::chrono::microseconds(interval));
501  } while(count < num_packets_);
502 
503  print_registers(std::cout, hafu);
504 
505  std::cout << std::endl;
506 
507  if (eth_ifc == "") {
508  std::cout << "No eth interface, so not "
509  "showing stats." << std::endl;
510  } else {
511  show_eth_stats(eth_ifc);
512 
513  if (eth_loopback_ == "on")
514  enable_eth_loopback(eth_ifc, false);
515  }
516 
517  if ((dfh.major_rev < 2) && (continuous_ == "on" || (contmonitor_ > 0)))
518  {
519  std::cout << "\nHSSI doesn't support continuous mode\n" << std::endl;
520  return test_afu::error;
521  }
522 
523  /* Monitor Implementation */
524 
525  if (continuous_ != "off" && contmonitor_ != 0)
526  {
527  uint32_t timer = 0;
528  uint32_t max_timer = contmonitor_;
529  char key;
530 
531  uint8_t port_size = (uint8_t)port_.size();
532  sort(port_.begin(), port_.end());
533  std::vector<perf_data*> old_perf_data(port_size); // old performance
534 
535  /* Initialise performance data and initiate monitor */
536  print_monitor_headers(std::cout, max_timer);
537  for(uint8_t port=0; port < port_size; port++)
538  {
539  old_perf_data[port] = (perf_data*)malloc(sizeof(perf_data));
540  select_port(port_[port], config_data, hafu);
541  capture_perf(old_perf_data[port], hafu); // captures the monitor data for first time
542  }
543 
544  /* Calculate performance and run monitor */
545  while(timer < max_timer)
546  {
547  for(uint8_t port=0; port < port_size; port++)
548  {
549  select_port(port_[port], config_data, hafu);
550  run_monitor(std::cout, old_perf_data[port], port_[port], hafu, timer+1) << std::endl;
551  }
552  move_cursor(port_size, CURSOR_UP);
553  timer+=1;
554 
555  key = get_user_input(key);
556  if(key=='q')
557  {
558  quit_monitor(port_size, CURSOR_DOWN, hafu);
559  return test_afu::success;
560  }
561  if(key=='r')
562  reset_monitor(port_, CURSOR_UP, config_data, hafu);
563 
564  //Handling CTL+C and CTL+Z
565  if (!running()) {
566  quit_monitor(port_size, CURSOR_DOWN, hafu);
567  return test_afu::error;
568  }
569  }
570  quit_monitor(port_size, CURSOR_DOWN, hafu);
571  }
572  else if (contmonitor_ > 0)
573  {
574  std::cout << "Please use --continuous and --contmonitor together" << std::endl;
575  return test_afu::error;
576  }
577  return test_afu::success;
578  }
