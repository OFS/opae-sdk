393 class host_exerciser : public test_afu {
394 public:
395  host_exerciser()
396  : test_afu("host_exerciser", nullptr, "info"), count_(1) {}
397 
398  virtual int run(CLI::App *app, test_command::ptr_t test) override {
399  int res = exit_codes::not_run;
400 
401  logger_->set_pattern(" %v");
402  // Info prints details of an individual run. Turn it on if doing only one
403  // test and the user hasn't changed level from the default.
404  if ((log_level_.compare("warning") == 0))
405  logger_->set_level(spdlog::level::info);
406 
407 
408  logger_->info("starting test run, count of {0:d}", count_);
409  uint32_t count = 0;
410  try {
411  while (count < count_) {
412  logger_->debug("starting iteration: {0:d}", count + 1);
413 
414  res = test_afu::run(app, test);
415  count++;
416  logger_->debug("end iteration: {0:d}", count);
417  if (res)
418  break;
419  }
420  } catch (std::exception &ex) {
421  logger_->error(ex.what());
422  res = exit_codes::exception;
423  }
424 
425  auto pass = res == exit_codes::success ? "PASS" : "FAIL";
426  logger_->info("Test {}({}): {}", test->name(), count, pass);
427  spdlog::drop_all();
428  return res;
429  }
430 
431 public:
432  uint32_t count_;
433 
434  bool option_passed(std::string option_str) {
435  if (app_.count(option_str) == 0)
436  return false;
437  return true;
438  }
439 };
