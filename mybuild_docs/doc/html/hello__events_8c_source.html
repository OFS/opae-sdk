76 struct ras_inject_error {

78  uint64_t csr;
79  struct {

83  uint64_t fatal_error : 1;

92 fpga_result inject_ras_fatal_error(fpga_token fme_token, uint8_t err)
93 {
94  fpga_result res1 = FPGA_OK;
95  fpga_result res2 = FPGA_OK;
96  fpga_handle fme_handle = NULL;
97  struct ras_inject_error inj_error = { {0} };
98  fpga_object inj_err_object;
99 
100  res1 = fpgaOpen(fme_token, &fme_handle, FPGA_OPEN_SHARED);
101  if (res1 != FPGA_OK) {
102  OPAE_ERR("Failed to open FPGA");
103  return res1;
104  }
105 
106  res1 = fpgaHandleGetObject(fme_handle, FME_SYSFS_INJECT_ERROR, &inj_err_object, 0);
107  ON_ERR_GOTO(res1, out_close, "Failed to get Handle Object");
108 
109  // Inject fatal error
110  inj_error.fatal_error = err;
111 
112  res1 = fpgaObjectWrite64(inj_err_object, inj_error.csr, 0);
113  ON_ERR_GOTO(res1, out_destroy_obj, "Failed to Read Object");
114 
115 out_destroy_obj:
116  res2 = fpgaDestroyObject(&inj_err_object);
117  ON_ERR_GOTO(res2, out_close, "Failed to Destroy Object");
118 out_close:
119  res2 = fpgaClose(fme_handle);
120  if (res2 != FPGA_OK) {
121  OPAE_ERR("Failed to close FPGA");
122  }
123 
124  return res1 != FPGA_OK ? res1 : res2;
125 }

165 fpga_result parse_args(int argc, char *argv[])
166 {
167  struct option longopts[] = {
168  { "help", no_argument, NULL, 'h' },
169  { "version", no_argument, NULL, 'v' },
170  { NULL, 0, NULL, 0 }
171  };
172 
173  int getopt_ret;
174  int option_index;
175 
176  while (-1 != (getopt_ret = getopt_long(argc, argv, GETOPT_STRING, longopts, &option_index))) {
177  const char *tmp_optarg = optarg;
178  /* checks to see if optarg is null and if not goes to value of optarg */
179  if ((optarg) && ('=' == *tmp_optarg)) {
180  ++tmp_optarg;
181  }
182 
183  switch (getopt_ret) {
184  case 'h': /* help */
185  help();
186  return -1;
187 
188  case 'v': /* version */
189  printf("hello_events %s %s%s\n",
190  OPAE_VERSION,
191  OPAE_GIT_COMMIT_HASH,
192  OPAE_GIT_SRC_TREE_DIRTY ? "*":"");
193  return -1;
194 
195  default: /* invalid option */
196  fprintf(stderr, "Invalid cmdline options\n");
197  return FPGA_EXCEPTION;
198  }
199  }
200 
201  return FPGA_OK;
202 }

204 fpga_result find_fpga(fpga_properties device_filter,
205  fpga_token *fpga,
206  uint32_t *num_matches)
207 {
208  fpga_properties filter = NULL;
209  fpga_result res = FPGA_OK;
210  fpga_result dres = FPGA_OK;
211 
212  res = fpgaCloneProperties(device_filter, &filter);
213  ON_ERR_GOTO(res, out, "cloning properties object");
214 
215  res = fpgaPropertiesSetObjectType(filter, FPGA_DEVICE);
216  ON_ERR_GOTO(res, out_destroy, "setting interface ID");
217 
218  res = fpgaEnumerate(&filter, 1, fpga, 1, num_matches);
219  ON_ERR_GOTO(res, out_destroy, "enumerating FPGAs");
220 
221 out_destroy:
222  dres = fpgaDestroyProperties(&filter);
223  ON_ERR_GOTO(dres, out, "destroying properties object");
224 out:
225  return (res == FPGA_OK) ? dres : res;
226 }

228 int main(int argc, char *argv[])
229 {
230  fpga_token fpga_device_token = NULL;
231  fpga_handle fpga_device_handle = NULL;
232  uint32_t num_matches = 1;
233  fpga_result res1 = FPGA_OK;
234  fpga_result res2 = FPGA_OK;
235  fpga_event_handle eh;
236  uint64_t count = 0;
237  int res;
238  struct pollfd pfd;
239  int timeout = 10000;
240  int poll_ret = 0;
241  ssize_t bytes_read = 0;
242  pthread_t errthr;
243  fpga_properties device_filter = NULL;
244 
245  res1 = fpgaGetProperties(NULL, &device_filter);
246  if (res1) {
247  print_err("failed to alloc properties", res1);
248  return res1;
249  }
250 
251  if (opae_set_properties_from_args(device_filter,
252  &res1,
253  &argc,
254  argv)) {
255  print_err("failed arg parse", res1);
256  res1 = FPGA_EXCEPTION;
257  goto out_exit;
258  } else if (res1) {
259  print_err("failed to set properties", res1);
260  goto out_exit;
261  }
262 
263  res1 = parse_args(argc, argv);
264  if ((int)res1 < 0)
265  goto out_exit;
266  ON_ERR_GOTO(res1, out_exit, "parsing arguments");
267 
268  res1 = find_fpga(device_filter, &fpga_device_token, &num_matches);
269  ON_ERR_GOTO(res1, out_exit, "finding FPGA accelerator");
270 
271  if (num_matches < 1) {
272  fprintf(stderr, "No matches for address provided.\n");
273  res1 = FPGA_NOT_FOUND;
274  goto out_exit;
275  }
276 
277  if (num_matches > 1) {
278  fprintf(stderr, "Found more than one suitable slot.\n");
279  }
280 
281  res1 = fpgaOpen(fpga_device_token, &fpga_device_handle, FPGA_OPEN_SHARED);
282  ON_ERR_GOTO(res1, out_destroy_tok, "opening accelerator");
283 
284  res1 = fpgaCreateEventHandle(&eh);
285  ON_ERR_GOTO(res1, out_close, "creating event handle");
286 
287  res1 = fpgaRegisterEvent(fpga_device_handle, FPGA_EVENT_ERROR, eh, 0);
288  ON_ERR_GOTO(res1, out_destroy_eh, "registering an FME event");
289 
290  printf("Waiting for interrupts now...\n");
291 
292  res = pthread_create(&errthr, NULL, error_thread, fpga_device_token);
293  if (res) {
294  printf("Failed to create error_thread.\n");
295  res1 = FPGA_EXCEPTION;
296  goto out_destroy_eh;
297  }
298 
299 
300  res1 = fpgaGetOSObjectFromEventHandle(eh, &pfd.fd);
301  ON_ERR_GOTO(res1, out_join, "getting file descriptor");
302 
303  pfd.events = POLLIN;
304  poll_ret = poll(&pfd, 1, timeout);
305 
306  if (poll_ret < 0) {
307  printf("Poll error errno = %s\n", strerror(errno));
308  res1 = FPGA_EXCEPTION;
309  goto out_join;
310  } else if (poll_ret == 0) {
311  printf("Poll timeout occurred\n");
312  res1 = FPGA_EXCEPTION;
313  goto out_join;
314  } else {
315  printf("FME Interrupt occurred\n");
316  bytes_read = opae_read(pfd.fd, &count, sizeof(count));
317  if (bytes_read <= 0)
318  printf("WARNING: error reading from poll fd: %s\n",
319  bytes_read < 0 ? strerror(errno) : "zero bytes read");
320  }
321 
322  res1 = fpgaUnregisterEvent(fpga_device_handle, FPGA_EVENT_ERROR, eh);
323  ON_ERR_GOTO(res1, out_join, "unregistering an FME event");
324 
325  printf("Successfully tested Register/Unregister for FME events!\n");
326 
327 out_join:
328  pthread_join(errthr, NULL);
329 
330 out_destroy_eh:
331  res2 = fpgaDestroyEventHandle(&eh);
332  ON_ERR_GOTO(res2, out_close, "deleting event handle");
333 
334 out_close:
335  res2 = fpgaClose(fpga_device_handle);
336  ON_ERR_GOTO(res2, out_destroy_tok, "closing accelerator");
337 
338 out_destroy_tok:
339  res2 = fpgaDestroyToken(&fpga_device_token);
340  ON_ERR_GOTO(res2, out_exit, "destroying token");
341 
342 out_exit:
343  fpgaDestroyProperties(&device_filter);
344  return res1 != FPGA_OK ? res1 : res2;
345 }
