83  void tg_perf (mem_tg *tg_exe_) {
84 
85  // Lock mutex before printing so print statements don't collide between threads.
86  std::unique_lock<std::mutex> print_lock(tg_print_mutex);
87  std::cout << "Channel " << std::stoi(tg_exe_->mem_ch_[0]) << ":" << std::endl;
88 
89  if (tg_exe_->status_ == TG_STATUS_TIMEOUT) {
90  std::cout << "TG TIMEOUT" << std::endl;

92  std::cout << "Error: Timed out in TG_STATUS_ACTIVE state. Consider increasing MEM_TG_TEST_TIMEOUT." << std::endl;
93  } else if (tg_exe_->status_ == TG_STATUS_ERROR) {
94  uint32_t tg_fail_exp;
95  uint32_t tg_fail_act;
96  uint64_t tg_fail_addr;
97  std::cout << "TG ERROR" << std::endl;

99  tg_fail_exp = tg_exe_->read64(tg_exe_->tg_offset_ + TG_FAIL_EXPECTED_DATA);
100  tg_fail_act = tg_exe_->read64(tg_exe_->tg_offset_ + TG_FAIL_READ_DATA);
101  std::cout << "Failed at address 0x" << std::hex << tg_fail_addr << " exp=0x" << tg_fail_exp << " act=0x" << tg_fail_act << std::endl;
102  } else {
103  std::cout << "TG PASS" << std::endl;
104  }
105 
106  uint32_t mem_ch_offset = (std::stoi(tg_exe_->mem_ch_[0])) << 0x3;
107  uint64_t num_ticks = tg_exe_->read64(MEM_TG_CLOCKS + mem_ch_offset);
108  std::cout << "Mem Clock Cycles: " << std::dec << num_ticks << std::endl;
109 
110  std::cout << "DEBUG: wcnt_ " << std::dec << tg_exe_->wcnt_ << std::endl;

124  bool tg_wait_test_completion (mem_tg *tg_exe_)
125  {
126  /* Wait for test completion */
127  uint32_t timeout = MEM_TG_TEST_TIMEOUT * tg_exe_->loop_ * tg_exe_->bcnt_;
128  // poll while active bit is set (channel status = {pass,fail,timeout,active})
129  uint32_t tg_status = 0x1;
130  tg_status = 0xF & (tg_exe_->read64(MEM_TG_STAT) >> (0x4*(std::stoi(tg_exe_->mem_ch_[0]))));
131  while (tg_status == TG_STATUS_ACTIVE) {
132  tg_status = 0xF & (tg_exe_->read64(MEM_TG_STAT) >> (0x4*(std::stoi(tg_exe_->mem_ch_[0]))));
133  std::this_thread::yield();
134  if (--timeout == 0) {
135  tg_exe_->status_ = -1;
136  std::cout << "DEBUG: timeout error" << std::endl;
137  return false;
138  }
139  }
140  if (tg_status == TG_STATUS_TIMEOUT) {
141  tg_exe_->status_ = tg_status;
142  std::cout << "DEBUG: status timeout" << std::endl;
143  return false;
144  }
145 
146  if (tg_status == TG_STATUS_ERROR) {
147  tg_exe_->status_ = tg_status;
148  std::cout << "DEBUG: status error" << std::endl;
149  return false;
150  }
151  tg_exe_->status_ = tg_status;
152  return true;
153  }

155  int config_input_options(mem_tg *tg_exe_)
156  {
157  if (!tg_exe_)
158  return -1;
159  uint64_t mem_capability = tg_exe_->read64(MEM_TG_CTRL);
160  if ((mem_capability & (0x1 << std::stoi(tg_exe_->mem_ch_[0]))) == 0) {
161  std::cerr << "No traffic generator for mem[" << std::stoi(tg_exe_->mem_ch_[0]) << "]" << std::endl;
162  return -1;
163  }
164 
165  tg_exe_->tg_offset_ = AFU_DFH + (MEM_TG_CFG_OFFSET * (1+std::stoi(tg_exe_->mem_ch_[0])));
166 
167  tg_exe_->write32(tg_exe_->tg_offset_+TG_LOOP_COUNT, tg_exe_->loop_);
168  tg_exe_->write32(tg_exe_->tg_offset_+TG_WRITE_COUNT, tg_exe_->wcnt_);
169  tg_exe_->write32(tg_exe_->tg_offset_+TG_READ_COUNT, tg_exe_->rcnt_);
170  tg_exe_->write32(tg_exe_->tg_offset_+TG_BURST_LENGTH, tg_exe_->bcnt_);
171  tg_exe_->write32(tg_exe_->tg_offset_+TG_SEQ_ADDR_INCR, tg_exe_->stride_);
172  tg_exe_->write32(tg_exe_->tg_offset_+TG_PPPG_SEL, tg_exe_->pattern_);
173 
174  // address increment mode
175  tg_exe_->write32(tg_exe_->tg_offset_+TG_ADDR_MODE_WR, TG_ADDR_SEQ);
176  tg_exe_->write32(tg_exe_->tg_offset_+TG_ADDR_MODE_RD, TG_ADDR_SEQ);
177  return 0;
178  }

180  // The test state has been configured. Run one test instance.
181  int run_mem_test(mem_tg *tg_exe_)
182  {
183  int status = 0;
184 
185  // All threads do their set up and wait here for other threads so start write happens all at once
186  std::unique_lock<std::mutex> lock(tg_start_write_mutex);
187  tg_waiting_threads_counter++;
188  if(tg_waiting_threads_counter == tg_num_threads) {
189  tg_cv.notify_all();
190  } else {
191  tg_cv.wait(lock, [&](){ return tg_waiting_threads_counter == tg_num_threads; });
192  }
193  lock.unlock();
194 
195  // write32 is not thread safe so we wrap it in a lock
196  lock.lock();
197  tg_exe_->logger_->debug("Start Test");
198  tg_exe_->write32(tg_exe_->tg_offset_ + TG_START, 0x1);
199  lock.unlock();
200 
201  if (!tg_wait_test_completion(tg_exe_))
202  status = -1;
203 
204  tg_perf(tg_exe_);
205 
206  return status;
207  }

218  virtual int run(test_afu *afu, CLI::App *app) override
219  {
220  (void)app;
221  auto d_afu = dynamic_cast<mem_tg *>(afu);
222  tg_exe_ = dynamic_cast<mem_tg *>(afu);
223 
224  token_ = d_afu->get_token();
225 
226  // Read HW details
227 
228  if (0 == tg_exe_->mem_speed_) {
229  tg_exe_->mem_speed_ = 300;
230  std::cout << "Memory channel clock frequency unknown. Assuming "
231  << tg_exe_->mem_speed_ << " MHz." << std::endl;
232  } else {
233  std::cout << "Memory clock from command line: "
234  << tg_exe_->mem_speed_ << " MHz" << std::endl;
235  }
236 
237  if (0 >= (tg_exe_->mem_ch_).size()) {
238  std::cout << "Insufficient arguments provided" << std::endl;
239  exit(1);
240  }
241 
242  // Parse mem_ch_ into array of selected channels and number of channels
243  int *channels = NULL;
244  int num_channels = 0;
245  if ((tg_exe_->mem_ch_[0]).find("all") == 0) {
246  uint64_t mem_capability = tg_exe_->read64(MEM_TG_CTRL);
247  channels = new int[sizeof(uint64_t)]; // size should be same as mem_capability
248  for (uint32_t i = 0; i < sizeof(uint64_t); i++) { // number of iterations should be same as mem_capability
249  if ((mem_capability & (1ULL << i)) != 0) {
250  channels[num_channels] = i;
251  num_channels += 1;
252  }
253  }
254  channels[num_channels] = -1; // EOL
255  } else {
256  channels = new int[tg_exe_->mem_ch_.size()];
257  num_channels = tg_exe_->mem_ch_.size();
258  try{
259  for (unsigned i = 0; i < tg_exe_->mem_ch_.size(); i++) {
260  channels[i] = std::stoi(tg_exe_->mem_ch_[i]);
261  }
262  } catch (std::invalid_argument &e) {
263  std::cerr << "Error: invalid argument to std::stoi";
264  delete[] channels;
265  return 1;
266  }
267  }
268 
269  // Spawn threads for each channel:
270  mem_tg *thread_tg_exe_objects[num_channels];
271  std::vector<std::future<int>> futures;
272  std::vector<std::promise<int>> promises(num_channels);
273  std::vector<std::thread> threads;
274  tg_num_threads = num_channels;
275  tg_waiting_threads_counter = 0;
276  for (int i = 0; i < num_channels; i++) {
277  if (channels[i] == -1) break;
278  thread_tg_exe_objects[i] = new mem_tg;
279  tg_exe_->duplicate(thread_tg_exe_objects[i]);
280  thread_tg_exe_objects[i]->mem_ch_.clear();
281  thread_tg_exe_objects[i]->mem_ch_.push_back(std::to_string(channels[i]));
282  futures.push_back(promises[i].get_future());
283  threads.emplace_back([&, i] {
284  promises[i].set_value(run_thread_single_channel(thread_tg_exe_objects[i]));
285  });
286  }
287 
288  // Wait for all threads to exit then collect their exit statuses
289  for (auto &thread : threads) {
290  thread.join();
291  }
292 
293  std::vector<int> exit_codes;
294  for (auto &future : futures) {
295  exit_codes.push_back(future.get());
296  }
297 
298  // Print message showing thread statuses
299  for (int i = 0; i < num_channels; i++) {
300  std::cout << "Thread on channel " << channels[i] << " exited with status " << (long)exit_codes[i] << std::endl;
301  }
302 
303  // Delete dynamic allocations
304  delete[] channels;
305  for (int i = 0; i < num_channels; i++) {
306  delete thread_tg_exe_objects[i];
307  }
308  return 0;
309  }
