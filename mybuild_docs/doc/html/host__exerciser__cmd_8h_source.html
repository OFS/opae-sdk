146  void he_perf_counters()
147  {
148  volatile he_dsm_status *dsm_status = NULL;
149  volatile uint8_t* status_ptr = dsm_->c_type();
150  uint64_t num_cache_lines = 0;
151  if (!status_ptr)
152  return;
153 
154  dsm_status = reinterpret_cast<he_dsm_status *>((uint8_t*)status_ptr);
155  if (!dsm_status)
156  return;
157 
158  host_exe_->logger_->info("Host Exerciser Performance Counter:");
159  // calculate number of cache lines in continuous mode
160  if (host_exe_->he_continuousmode_) {
161  if (he_lpbk_cfg_.TestMode == HOST_EXEMODE_READ)
162  num_cache_lines = dsm_num_reads(dsm_status);
163  else if (he_lpbk_cfg_.TestMode == HOST_EXEMODE_WRITE)
164  num_cache_lines = (dsm_num_writes(dsm_status));
165  else
166  num_cache_lines = dsm_num_reads(dsm_status) + dsm_num_writes(dsm_status);
167  } else {
168  num_cache_lines = (LPBK1_BUFFER_SIZE / (1 * CL));
169  }
170 
171  host_exerciser_status();
172 
173  uint64_t tmp;
174 
175  tmp = dsm_num_ticks(dsm_status);
176  host_exe_->logger_->info("Number of clocks: {0}", tmp);
177  tmp = dsm_num_reads(dsm_status);
178  host_exe_->logger_->info("Total number of Reads sent: {0}", tmp);
179  tmp = dsm_num_writes(dsm_status);
180  host_exe_->logger_->info("Total number of Writes sent: {0}", tmp);
181 
182  // print bandwidth
183  if (dsm_num_ticks(dsm_status) > 0) {
184  double perf_data = he_num_xfers_to_bw(num_cache_lines, dsm_num_ticks(dsm_status));
185  host_exe_->logger_->info("Bandwidth: {0:0.3f} GB/s", perf_data);

243  void he_init_src_buffer(shared_buffer::ptr_t buffer)
244  {
245  // Fill the source buffer with random values
246  host_exe_->fill(buffer);
247 
248  // Compare and swap? If so, seed the source buffers with values will
249  // match. The hardware tests uses the line index as the test.
250  if (he_lpbk_cfg_.AtomicFunc == HOSTEXE_ATOMIC_CAS_4) {
251  for (uint32_t i = 0; i < buffer->size()/CL; i += 3) {
252  buffer->write<uint32_t>(i, i*CL);
253  }
254  }
255  if (he_lpbk_cfg_.AtomicFunc == HOSTEXE_ATOMIC_CAS_8) {
256  for (uint32_t i = 0; i < buffer->size()/CL; i += 3) {
257  buffer->write<uint64_t>(i, i*CL);
258  }
259  }
260 
261  // In atomic mode, at most the first 8 bytes of each line will be
262  // updated and copied. In the source buffer, write a function of
263  // the value at the start of each line to the second position so
264  // it can be used as a check later.
265  if (he_lpbk_cfg_.AtomicFunc != HOSTEXE_ATOMIC_OFF) {
266  for (uint32_t i = 0; i < buffer->size()/CL; i += 1) {
267  uint64_t v = buffer->read<uint64_t>(i*CL);
268  buffer->write<uint64_t>(v ^ 0xabababababababab, i*CL + 8);
269  }
270  }
271  }

295  uint64_t he_compare_buffer()
296  {
297  uint64_t status = host_exerciser_swtestmsg();
298  if (status) {
299  std::cerr << "HE LB CSR error flag is set" << std::endl;
300  return status;
301  }
302 
303  // Compare buffer contents only loopback test mode
304  if (he_lpbk_cfg_.TestMode != HOST_EXEMODE_LPBK1)
305  return 0;
306 
307  // Normal (non-atomic) is a simple comparison
308  if (he_lpbk_cfg_.AtomicFunc == HOSTEXE_ATOMIC_OFF) {
309  return (source_->compare(destination_, source_->size()));
310  }
311 
312  // Atomic mode is a far more complicated comparison. The source buffer
313  // is modified and some of the original state is copied to the destination.
314  bool size_is_4 = ((he_lpbk_cfg_.AtomicFunc & 2) == 0);
315  bool is_fetch_add = (he_lpbk_cfg_.AtomicFunc & 0xc) == 0;
316  bool is_swap = (he_lpbk_cfg_.AtomicFunc & 0xc) == 4;
317  bool is_cas = (he_lpbk_cfg_.AtomicFunc & 0xc) == 8;
318 
319  // Ignore the last entry to work around an off by one copy error
320  for (uint64_t i = 0; i < source_->size()/CL; i += 1) {
321  // In source_, the first entry in every line is the atomically modified
322  // value. The second entry holds the original value, hashed with a constant
323  // so it isn't a simple repetition.
324  uint64_t src_a = source_->read<uint64_t>(i*CL);
325  // Read the original value and reverse the hash.
326  uint64_t src_b = source_->read<uint64_t>(i*CL + 8) ^ 0xabababababababab;
327  uint64_t upd_a = 0;
328 
329  // Compute expected value
330  if (is_fetch_add) {
331  // Hardware added the line index (i) to each line, either preserving
332  // the high 4 bytes or modifying all 8 bytes.
333  if (size_is_4)
334  upd_a = (src_b & 0xffffffff00000000) | ((src_b + i) & 0xffffffff);
335  else
336  upd_a = src_b + i;
337  }
338  if (is_swap) {
339  // Hardware swapped the line index (i) into each line
340  if (size_is_4)
341  upd_a = (src_b & 0xffffffff00000000) | (i & 0xffffffff);
342  else
343  upd_a = i;
344  }
345  if (is_cas) {
346  // Hardware swapped the bit inverse of line index (i) in each line when
347  // the original value is the line index.
348  if (size_is_4)
349  upd_a = ((src_b & 0xffffffff) == i) ? (src_b & 0xffffffff00000000) | (~i & 0xffffffff) : src_b;
350  else
351  upd_a = (src_b == i) ? ~i : src_b;
352  }
353 
354  if (upd_a != src_a) {
355  std::cerr << "Atomic update error" << std::endl;
356  return 1;
357  }
358 
359  // The destination is comparatively easy. For all functions it is
360  // simply the original source value.
361  uint64_t dst_a = destination_->read<uint64_t>(i*CL);
362  if (size_is_4) {
363  src_b &= 0xffffffff;
364  dst_a &= 0xffffffff;
365  }
366 
367  if (dst_a != src_b) {
368  std::cerr << "Atomic read error or write error" << std::endl;
369  return 1;
370  }
371  }
372 
373  return 0;
374  }

415  int parse_input_options()
416  {
417 
418  if (!host_exe_)
419  return -1;
420 
421  // ASE simulation?
422  auto afu_props = fpga::properties::get(token_);
423  uint32_t vendor_id = afu_props->vendor_id;
424  uint32_t device_id = afu_props->device_id;
425  is_ase_sim_ = (vendor_id == 0x8086) && (device_id == 0xa5e);
426  if (is_ase_sim_) {
427  std::cout << "Simulation: ASE mode" << std::endl;
428  }
429 
430  // Host Exerciser Mode
431  he_lpbk_cfg_.TestMode = host_exe_->he_modes_;
432 
433  // Host Exerciser Read
434  if (host_exe_->he_req_cls_len_ > he_lpbk_max_reqlen_) {
435  std::cerr << "Request length " << host_exe_->he_req_cls_len_
436  << " is not supported by this platform." << std::endl;
437  return -1;
438  }
439  set_cfg_reqlen(host_exe_->he_req_cls_len_);
440 
441  // Host Exerciser lpbk delay
442  if (host_exe_->he_delay_)
443  he_lpbk_cfg_.DelayEn = 1;
444 
445  //test rollover or test termination
446  if (host_exe_->he_continuousmode_)
447  he_lpbk_cfg_.Continuous = 1;
448 
449  // Set interleave in Throughput
450  if (he_lpbk_cfg_.TestMode == HOST_EXEMODE_TRPUT) {
451  he_lpbk_cfg_.TputInterleave = host_exe_->he_interleave_;
452  }
453 
454  // Set Interrupt test mode
455  if (host_exe_->option_passed("--interrupt")) {
456  if (host_exe_->he_interrupt_ < afu_props->num_interrupts) {
457  he_lpbk_cfg_.IntrTestMode = 1;
458  } else {
459  std::cerr << "Invalid Input interrupts vector number:" << host_exe_->he_interrupt_ << std::endl;
460  std::cout << "Please enter Interrupts vector range 0 to " << afu_props->num_interrupts -1 << std::endl;
461  return -1;
462  }
463  }
464 
465  // Atomic functions
466  he_lpbk_cfg_.AtomicFunc = host_exe_->he_req_atomic_func_;
467  if (he_lpbk_cfg_.AtomicFunc != HOSTEXE_ATOMIC_OFF) {
468  if (!he_lpbk_atomics_supported_) {
469  std::cerr << "The platform does not support atomic functions" << std::endl;
470  return -1;
471  }
472  if ((get_cfg_reqlen() != HOSTEXE_CLS_1) && (he_lpbk_cfg_.TestMode == HOST_EXEMODE_LPBK1)) {
473  std::cerr << "Atomic function in lpbk mode requires cl_1" << std::endl;
474  return -1;
475  }
476  }
477 
478  // Encoding
479  he_lpbk_cfg_.Encoding = host_exe_->he_req_encoding_;
480  if ((he_lpbk_cfg_.Encoding != HOSTEXE_ENCODING_DEFAULT) && (he_lpbk_api_ver_ == 0)) {
481  std::cerr << "Host exerciser hardware API version 0 does not support encoding changes" << std::endl;
482  return -1;
483  }
484 
485  if (host_exe_->he_continuousmode_ &&
486  (he_lpbk_cfg_.IntrTestMode == 1)) {
487  std::cerr << "Interrupts not supported in continuous mode"
488  << std::endl;
489  return -1;
490  }
491 
492  if (host_exe_->he_continuousmode_ &&
493  (host_exe_->he_contmodetime_ == 0)) {
494  std::cerr << "Please add cmd line input continuous mode time"
495  << std::endl;
496  return -1;
497  }
498 
499  return 0;
500  }

502  // The test state has been configured. Run one test instance.
503  int run_single_test()
504  {
505  int status = 0;
506 
507  // Clear the status buffer
508  std::fill_n(dsm_->c_type(), LPBK1_DSM_SIZE, 0x0);
509 
510  host_exe_->logger_->debug("Start Test");
511 
512  // Write to CSR_CFG
513  host_exe_->write64(HE_CFG, he_lpbk_cfg_.value);
514  host_exe_->logger_->debug("Input Config: {0}", he_lpbk_cfg_.value);
515 
516  event::ptr_t ev = nullptr;
517  if (he_lpbk_cfg_.IntrTestMode == 1) {
518  he_interrupt_.VectorNum = host_exe_->he_interrupt_;
519  host_exe_->write32(HE_INTERRUPT0, he_interrupt_.value);
520  ev = host_exe_->register_interrupt(host_exe_->he_interrupt_);
521  std::cout << "Using Interrupts\n";
522  }
523 
524  if (host_exe_->logger_->should_log(spdlog::level::debug)) {
525  std::cout << std::endl;
526  he_dump_buffer(source_, "Pre-execution source");
527  he_dump_buffer(destination_, "Pre-execution destination");
528  std::cout << std::endl;
529  }
530 
531  // Write to CSR_CTL
532  he_lpbk_ctl_.value = 0;
533  he_lpbk_ctl_.Start = 1;
534  he_lpbk_ctl_.ResetL = 1;
535  host_exe_->write32(HE_CTL, he_lpbk_ctl_.value);
536 
537  // Interrupt test mode
538  if (he_lpbk_cfg_.IntrTestMode == 1) {
539  if (!he_interrupt(ev)) {
540  status = -1;
541  if (host_exe_->logger_->should_log(spdlog::level::debug)) {
542  std::cout << std::endl;

544  he_dump_buffer(destination_, "Post-execution destination");
545  std::cout << std::endl;
546  }
547  }
548  else {
549  status = he_compare_buffer();
550  if (host_exe_->logger_->should_log(spdlog::level::debug)) {

553  he_dump_buffer(destination_, "Post-execution destination");
554  std::cout << std::endl;
555  }
556  he_perf_counters();
557  }
558  } else if (host_exe_->he_continuousmode_) {
559  // Continuous mode
560  he_continuousmode();
561 
562  if (host_exe_->logger_->should_log(spdlog::level::debug)) {
563  std::cout << std::endl;
564  he_dump_buffer(source_, "Post-execution source");
565  he_dump_buffer(destination_, "Post-execution destination");
566  std::cout << std::endl;
567  }
568  } else {
569  // Regular mode
570  if (!he_wait_test_completion()) {
571  status = -1;
572  }
573  else {
574  if (host_exe_->logger_->should_log(spdlog::level::debug)) {
575  std::cout << std::endl;
576  he_dump_buffer(source_, "Post-execution source");
577  he_dump_buffer(destination_, "Post-execution destination");
578  std::cout << std::endl;
579  }
580 
581  status = he_compare_buffer();
582  he_perf_counters();
583  }
584  }
585 
586  // assert reset he-lpbk
587  he_lpbk_ctl_.value = 0;
588  host_exe_->write32(HE_CTL, he_lpbk_ctl_.value);
589  usleep(1000);
590 
591  // deassert reset he-lpbk
592  he_lpbk_ctl_.value = 0;
593  he_lpbk_ctl_.ResetL = 1;
594  host_exe_->write32(HE_CTL, he_lpbk_ctl_.value);
595 
596  return status;

599  // Sequence through all the test modes
600  int run_all_tests()
601  {
602  int status = 0;
603 
604  // Start with loopback tests
