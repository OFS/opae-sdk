60  bool glob_ethernet_interface(const std::string &glob_pattern, std::string &eth_ifc)
61  {
62  glob_t gl;
63 
64  if (glob(glob_pattern.c_str(), 0, nullptr, &gl)) {
65  if (gl.gl_pathv)
66  globfree(&gl);
67  return false;
68  }
69 
70  if (!gl.gl_pathc) {
71  if (gl.gl_pathv)

78 
79  std::string ifc(gl.gl_pathv[0]);
80  size_t pos = ifc.rfind("/") + 1;
81  eth_ifc = ifc.substr(pos);

126  void mbox_write(uint16_t offset, uint32_t data)
127  {
128  volatile uint8_t *mmio_base = handle_->mmio_ptr(0);
129 
130  uint64_t val;
131  struct timespec ts;
132  uint64_t ticks;
133  const uint64_t max_ticks = 10000ULL;
134 
135  val = (((uint64_t)data) << WRITE_DATA_SHIFT);
136  *((volatile uint64_t *)(mmio_base + TRAFFIC_CTRL_DATA)) = val;
137 
138  val = (((uint64_t)offset) << AFU_CMD_SHIFT) | WRITE_CMD;
139  *((volatile uint64_t *)(mmio_base + TRAFFIC_CTRL_CMD)) = val;
140 
141  ticks = max_ticks;
142  ts.tv_sec = 0;
143  ts.tv_nsec = 100;
144  do
145  {
146  val = *((volatile uint64_t *)(mmio_base + TRAFFIC_CTRL_CMD));
147  if (val & ACK_TRANS)
148  break;
149  if (nanosleep(&ts, NULL) != -1 &&
150  ticks != NO_TIMEOUT) {
151  if (!ticks) {
152  const char *msg = "mbox_write timed out [a]";
153  std::cerr << msg << std::endl;
154  throw std::runtime_error(msg);
155  }
156  --ticks;
157  }
158  } while (!(val & ACK_TRANS));
159 
160  ticks = max_ticks;
161  do
162  {
163  *((volatile uint64_t *)(mmio_base + TRAFFIC_CTRL_CMD)) = ACK_TRANS;
164  val = *((volatile uint64_t *)(mmio_base + TRAFFIC_CTRL_CMD));
165  if (!(val & ACK_TRANS))
166  break;
167  if (nanosleep(&ts, NULL) != -1 &&

171  std::cerr << msg << std::endl;

175  }
176  } while (val & ACK_TRANS);
177  }
