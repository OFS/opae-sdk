106 metric_group *init_metric_group(fpga_token token, const char *name,
107  metric_group *group)
108 {
109  fpga_properties props;
110  fpga_result res;
111 
112  group->token = token;
113  group->name = name;
114  group->count = 0;
115 
116  if (fpgaGetProperties(token, &props) == FPGA_OK) {
117  res = fpgaPropertiesGetBus(props, &group->bus);
118  if (res != FPGA_OK)
119  print_err("error reading bus", res);
120 
121  res = fpgaPropertiesGetDevice(props, &group->device);
122  if (res != FPGA_OK)
123  print_err("error reading device", res);
124 
125  res = fpgaPropertiesGetFunction(props, &group->function);
126  if (res != FPGA_OK)
127  print_err("error reading function", res);
128  }
129 
130  fpgaDestroyProperties(&props);
131 
132  if (fpgaTokenGetObject(token, name, &group->object, FPGA_OBJECT_GLOB)
133  == FPGA_OK) {
134  return group;
135  }
136 
137  return NULL;
138 }

152 fpga_result add_counter(metric_group *group, const char *name)
153 {
154  fpga_result res = FPGA_EXCEPTION;
155  size_t count = group->count;
156  res = fpgaObjectGetObject(group->object, name,
157  &group->objects[count].object,
158  FPGA_OBJECT_GLOB);
159  if (res == FPGA_OK) {
160  group->objects[count].value = 0;
161  group->objects[count].name = name;
162  group->count++;
163  ADD_TO_CLEANUP(fpgaDestroyObject, &group->objects[count].object);
164 
165  } else {
166  group->objects[count].object = NULL;
167  }
168  return res;
169 }

171 void print_counters(fpga_object clock, metric_group *group)
172 {
173  size_t count = group->count;
174  size_t i;
175  uint64_t value = 0, clock_value = 0;
176  fpga_result res = fpgaObjectRead64(clock, &clock_value, FPGA_OBJECT_SYNC);
177  if (res != FPGA_OK) {
178  print_err("Error reading clock", res);
179  return;
180  }
181  for (i = 0; i < count; ++i) {
182  res = fpgaObjectRead64(group->objects[i].object, &value,
183  FPGA_OBJECT_SYNC);
184  if (res == FPGA_OK) {
185  group->objects[i].delta =
186  value - group->objects[i].value;
187  group->objects[i].value = value;
188  } else {
189  print_err("error reading counter", res);
190  }
191  }
192  printf("device: %02x:%02x.%1d clock %lu: \n\t", group->bus,
193  group->device, group->function, clock_value);
194  for (i = 0; i < count; ++i) {
195  if (group->objects[i].delta > 0)
196  printf("%16s : %10lu, ", group->objects[i].name,
197  group->objects[i].delta);
198  }
199 
200  printf("\n");
201 }

218 fpga_result parse_args(int argc, char *argv[])
219 {
220  struct option longopts[] = {
221  { "help", no_argument, NULL, 'h' },
222  { "version", no_argument, NULL, 'v' },
223  { NULL, 0, NULL, 0 }
224  };
225  int getopt_ret;
226  int option_index;
227 
228  while (-1 != (getopt_ret = getopt_long(argc, argv, GETOPT_STRING,
229  longopts, &option_index))) {
230  const char *tmp_optarg = optarg;
231  /* Checks to see if optarg is null and if not it goes to value
232  * of optarg */
233  if ((optarg) && ('=' == *tmp_optarg)) {
234  ++tmp_optarg;
235  }
236 
237  switch (getopt_ret) {
238  case 'h': /* help */
239  help();
240  return -1;
241 
242  case 'v': /* version */
243  printf("object_api %s %s%s\n",
244  OPAE_VERSION,
245  OPAE_GIT_COMMIT_HASH,
246  OPAE_GIT_SRC_TREE_DIRTY ? "*":"");
247  return -1;
248 
249  default: /* invalid option */
250  fprintf(stderr, "Invalid cmdline option \n");
251  return FPGA_EXCEPTION;
252  }
253  }
254 
255  return FPGA_OK;
256 }

260 int main(int argc, char *argv[])
261 {
262  fpga_properties filter;
263  fpga_token tokens[MAX_TOKENS];
264  fpga_result res = FPGA_OK;
265  int exit_code = -1;
266  uint32_t num_matches = 0;
267  uint32_t i = 0, j = 0;
268  token_group *metrics = NULL;
269 
270  res = fpgaGetProperties(NULL, &filter);
271  ON_ERR_GOTO(res, out, "Creating fpga_properties");
272  ADD_TO_CLEANUP(fpgaDestroyProperties, &filter);
273 
274  if (opae_set_properties_from_args(filter,
275  &res,
276  &argc,
277  argv)) {
278  print_err("failed arg parse", res);
279  goto out_free;
280  } else if (res) {
281  print_err("failed to set properties", res);
282  goto out_free;
283  }
284 
285  res = parse_args(argc, argv);
286  if (res != FPGA_OK) {
287  if ((int)res > 0)
288  print_err("error parsing arguments", res);
289  goto out_free;
290  }
291 
292  // set filter to accelerator object
293  res = fpgaPropertiesSetObjectType(filter, FPGA_DEVICE);
294  ON_ERR_GOTO(res, out_free, "Setting ObjectType");
295 
296  res = fpgaEnumerate(&filter, 1, tokens, MAX_TOKENS, &num_matches);
297  ON_ERR_GOTO(res, out_free, "Enumerating for properties");
298  for (i = 0; i < num_matches; ++i) {
299  ADD_TO_CLEANUP(fpgaDestroyToken, &tokens[i]);
300  }
301 
302  metrics = opae_calloc(num_matches, sizeof(token_group));
303  if (metrics == NULL) {
304  fprintf(stderr, "Could not allocate array for metrics\n");
305  goto out_free;
306  }
307 
308  for (i = 0; i < num_matches; ++i) {
309  metrics[i].token = tokens[i];
310  add_clock(&metrics[i]);
311  metrics[i].groups = opae_calloc(3, sizeof(metric_group));
312  metrics[i].count = 1;
313  metric_group *g_fabric = init_metric_group(
314  tokens[i], "*perf/fabric", &metrics[i].groups[0]);
315  if (!g_fabric)
316  goto out_free;
317  ADD_TO_CLEANUP(fpgaDestroyObject, &(g_fabric->object));
318  add_counter(g_fabric, "mmio_write");
319  add_counter(g_fabric, "mmio_read");
320  }
321  int count = 10;
322  while (--count > 0) {
323  for (i = 0; i < num_matches; ++i) {
324  for (j = 0; j < metrics[i].count; ++j) {
325  print_counters(metrics[i].clock,
326  &metrics[i].groups[j]);
327  }
328  usleep((useconds_t)(options.interval_sec
329  * USEC_TO_SEC));
330  }
331  }
332 
333  exit_code = 0;
334 
335 out_free:
336 
337  while (cleanup_size-- > 0) {
338  res = cleanup[cleanup_size].fn(cleanup[cleanup_size].param);
339  if (res != FPGA_OK) {
340  print_err("Error destroying structure: ", res);
341  }
342  }
343 
344  if (metrics) {
345  for (i = 0; i < num_matches; ++i) {
346  if (metrics[i].groups)
347  opae_free(metrics[i].groups);
348  }
349  opae_free(metrics);
350  }
351 out:
352  return exit_code;
353 }
