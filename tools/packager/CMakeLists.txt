# Generate executable python zip file
zip_files(
  ${CMAKE_CURRENT_BINARY_DIR}/packager.zip
  ${CMAKE_CURRENT_SOURCE_DIR}
  "*.py *.pyc metadata schema")

python_wrapped_exe(${PACKAGER_BIN} ${CMAKE_CURRENT_BINARY_DIR}/packager.zip)

# Install executable python zip file
install(
  PROGRAMS ${PACKAGER_BIN}
  DESTINATION bin
  COMPONENT toolpackager)
